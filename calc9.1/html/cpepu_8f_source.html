496  SUBROUTINE pleph ( ET, NTARG, NCENT, RRD )
497  Implicit None
498 C The following code is from JPL with only a few necessary modifications:
499 C*************************************************************************
500 C
501 C NOTE : Over the years, different versions of PLEPH have had a fifth argument:
502 C sometimes, an error return statement number; sometimes, a logical denoting
503 C whether or not the requested date is covered by the ephemeris. We apologize
504 C for this inconsistency; in this present version, we use only the four
505 C necessary arguments and do the testing outside of the subroutine.
506 C
507 C THIS SUBROUTINE READS THE JPL PLANETARY EPHEMERIS AND GIVES THE
508 C POSITION AND VELOCITY OF THE POINT 'NTARG' WITH RESPECT TO 'NCENT'.
509 C
510 C CALLING SEQUENCE PARAMETERS:
511 C
512 C ET = D.P. JULIAN EPHEMERIS DATE AT WHICH INTERPOLATION IS WANTED.
513 C
514 C ** NOTE THE ENTRY DPLEPH FOR A DOUBLY-DIMENSIONED TIME **
515 C THE REASON FOR THIS OPTION IS DISCUSSED IN THE SUBROUTINE STATE
516 C
517 C NTARG = INTEGER NUMBER OF 'TARGET' POINT.
518 C NCENT = INTEGER NUMBER OF CENTER POINT.
519 C
520 C THE NUMBERING CONVENTION FOR 'NTARG' AND 'NCENT' IS:
521 C 1 = MERCURY 8 = NEPTUNE
522 C 2 = VENUS 9 = PLUTO
523 C 3 = EARTH 10 = MOON
524 C 4 = MARS 11 = SUN
525 C 5 = JUPITER 12 = SOLAR-SYSTEM BARYCENTER
526 C 6 = SATURN 13 = EARTH-MOON BARYCENTER
527 C 7 = URANUS 14 = NUTATIONS (LONGITUDE AND OBLIQ)
528 C 15 = LIBRATIONS, IF ON EPH FILE

667  SUBROUTINE interp(BUF,T,NCF,NCM,NA,IFL,PV)
668  Implicit None
669 C
670 C THIS SUBROUTINE DIFFERENTIATES AND INTERPOLATES A
671 C SET OF CHEBYSHEV COEFFICIENTS TO GIVE POSITION AND VELOCITY
672 C
673 C CALLING SEQUENCE PARAMETERS:
674 C
675 C INPUT:
676 C BUF 1ST LOCATION OF ARRAY OF D.P. CHEBYSHEV COEFFICIENTS OF POSITION
677 C T T(1) IS DP FRACTIONAL TIME IN INTERVAL COVERED BY
678 C COEFFICIENTS AT WHICH INTERPOLATION IS WANTED
679 C (0 .LE. T(1) .LE. 1). T(2) IS DP LENGTH OF WHOLE
680 C INTERVAL IN INPUT TIME UNITS.
681 C NCF # OF COEFFICIENTS PER COMPONENT
682 C NCM # OF COMPONENTS PER SET OF COEFFICIENTS
683 C NA # OF SETS OF COEFFICIENTS IN FULL ARRAY
684 C (I.E., # OF SUB-INTERVALS IN FULL INTERVAL)
685 C IFL INTEGER FLAG: =1 FOR POSITIONS ONLY
686 C =2 FOR POS AND VEL
687 C
688 C OUTPUT:
689 C PV INTERPOLATED QUANTITIES REQUESTED. DIMENSION
690 C EXPECTED IS PV(NCM,IFL), DP.
691 C
692  SAVE
693  INTEGER*4 NCF, NCM, NA, IFL, L, NP, NV, I, J
694  REAL*8 BUF(ncf,ncm,*), T(2), PV(ncm,*), PC(18), VC(18),
695  * dna, dt1, temp, tc, twot, vfac
696 C
697  DATA np/2/
698  DATA nv/3/
699  DATA twot/0.d0/
700  DATA pc(1),pc(2)/1.d0,0.d0/
701  DATA vc(2)/1.d0/
702 C
703 C ENTRY POINT. GET CORRECT SUB-INTERVAL NUMBER FOR THIS SET
704 C OF COEFFICIENTS AND THEN GET NORMALIZED CHEBYSHEV TIME
705 C WITHIN THAT SUBINTERVAL.
706 C
707  dna=dble(na)
708  dt1=dint(t(1))
709  temp=dna*t(1)
710  l=idint(temp-dt1)+1
711 
712 C TC IS THE NORMALIZED CHEBYSHEV TIME (-1 .LE. TC .LE. 1)
713  tc=2.d0*(dmod(temp,1.d0)+dt1)-1.d0
714 C
715 C CHECK TO SEE WHETHER CHEBYSHEV TIME HAS CHANGED,
716 C AND COMPUTE NEW POLYNOMIAL VALUES IF IT HAS.
717 C (THE ELEMENT PC(2) IS THE VALUE OF T1(TC) AND HENCE
718 C CONTAINS THE VALUE OF TC ON THE PREVIOUS CALL.)
719 C
720  IF(tc.NE.pc(2)) THEN
721  np=2
722  nv=3
723  pc(2)=tc
724  twot=tc+tc
725  ENDIF
726 C
727 C BE SURE THAT AT LEAST 'NCF' POLYNOMIALS HAVE BEEN EVALUATED
728 C AND ARE STORED IN THE ARRAY 'PC'.
729  IF(np.LT.ncf) THEN
730  DO 1 i=np+1,ncf
731  pc(i)=twot*pc(i-1)-pc(i-2)
732  1 CONTINUE
733  np=ncf
734  ENDIF
735 C
736 C INTERPOLATE TO GET POSITION FOR EACH COMPONENT
737  DO 2 i=1,ncm
738  pv(i,1)=0.d0
739  DO 3 j=ncf,1,-1
740  pv(i,1)=pv(i,1)+pc(j)*buf(j,i,l)
741  3 CONTINUE
742  2 CONTINUE
743  IF(ifl.LE.1) RETURN
744 C
745 C IF VELOCITY INTERPOLATION IS WANTED, BE SURE ENOUGH
746 C DERIVATIVE POLYNOMIALS HAVE BEEN GENERATED AND STORED.
747  vfac=(dna+dna)/t(2)
748  vc(3)=twot+twot
749  IF(nv.LT.ncf) THEN
750  DO 4 i=nv+1,ncf
751  vc(i)=twot*vc(i-1)+pc(i-1)+pc(i-1)-vc(i-2)
752  4 CONTINUE
753  nv=ncf
754  ENDIF
755 C
756 C INTERPOLATE TO GET VELOCITY FOR EACH COMPONENT
757  DO 5 i=1,ncm
758  pv(i,2)=0.d0
759  DO 6 j=ncf,2,-1
760  pv(i,2)=pv(i,2)+vc(j)*buf(j,i,l)
761  6 CONTINUE

763  5 CONTINUE

765  RETURN
