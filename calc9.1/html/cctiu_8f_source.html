193  SUBROUTINE ctimg (AT, CFSITE, SITLON, UTC, XJD, CT, DATDCT,DLPGR)
194  IMPLICIT None
195 C
196 C 1. CTIMG
197 C
198 C 1.1 CTIMG PROGRAM SPECIFICATION
199 C
200 C 1.1.1 CTIMG is the utility routine which computes the coordinate time fraction
201 C of the coordinate time day at observing site #1, the partial derivative
202 C of the atomic time with respect to the coordinate time, and the partial
203 C derivative of the long period terms in the 'AT mimus CT' offset with
204 C respect to the coordinate time.
205 C
206 C The algorithm is taken directly from the referenced Moyer paper. Moyer
207 C derives the theory in terms of vector positions of the Earth, Moon, and
208 C planets in a solar system barycenter coordinate system and he gives an
209 C algorithm in that system. In part 2 of the paper he derives an alternate
210 C algorithm which does not use vectors, but rather uses an analytic
211 C planetary theory. The advantage of the second algorithm is that the only
212 C required inputs are the geocentric, crust fixed position of the station
213 C and the UTC and AT time tags. The second algorithm is implemented here.
214 C
215 C NOTE 1. The algorithm calls for UT1, but UTC is used in its' place.
216 C (Since 1972 UTC has been kept within 0.7 seconds of UT1).
217 C
218 C NOTE 2. The equation numbers in the Moyer paper are identified below.
219 C
220 C 1.1.3 REFERENCES - MOYER, T. D., TRANSFORMATION FROM PROPER TIME ON
221 C EARTH TO COORDINATE TIME IN SOLAR SYSTEM BARYCENTRIC
222 C SPACE-TIME FRAME OF REFERENCE, CEL. MECH. 23,
223 C 33-68, 1981. SEE PP. 66-67.
224 C
225 C 1.2 CTIMG PROGRAM INTERFACE
226 C
227 C 1.2.1 CALLING SEQUENCE -
228 C INPUT VARIABLES:
229 C 1. AT - THE ATOMIC TIME FRACTION OF THE TAI TIME DAY.
230 C (DAYS)
231 C 2. CFSITE(3,2) - CRUST FIXED STATION COORDINATES. (M)
232 C 3. SITLON(2) - STATION EAST LONGITUDES. (RAD)
233 C 4. UTC - UTC FRACTION OF DAY. (DAYS)
234 C 5. XJD - THE JULIAN DATE AT ZERO HOURS UTC OF THE DATE
235 C IN QUESTION. (DAYS)
236 C OUTPUT VARIABLES:
237 C 1. CT - THE COORDINATE TIME FRACTION OF THE COORDINATE TIME
238 C DAY. (DAYS)
239 C 2. DATDCT - THE PARTIAL DERIVATIVE OF THE ATOMIC TIME WITH
240 C RESPECT TO THE COORDINATE TIME. (SEC/SEC)
241 C 3. DLPGR - THE PARTIAL DERIVATIVE OF THE LONG PERIOD TERMS IN
242 C THE 'AT MINUS CT' OFFSET WITH RESPECT TO THE
243 C COORDINATE TIME. (SEC/SEC)
244 C
245 C 1.2.2 COMMON BLOCKS USED -
246 C
247  Real*8 A1TAI, ATCTEP, D1950, ECCEN, XL(2), XM(2),
248  . d(2), xllj(2), xllsa(2), xmj(2), xmsa(2), act(15)
249  COMMON / cticm / a1tai, atctep, d1950, eccen, xl, xm,
250  . d, xllj, xllsa, xmj, xmsa, act
251 C
252 C VARIABLES 'FROM':
253 C 1. A1TAI - THE OFFSET BETWEEN A1 AND TAI (SECONDS) (NO LONGER USED)
254 C 2. ATCTEP - THE VALUE OF THE 'CT MINUS TAI' OFFSET. (SEC)
255 C 3. D1950 - THE JULIAN DATE OF THE EPOCH JAN 1, 1950 0 HR UT.
256 C 4. ECCEN - ECCENTRICITY OF THE HELIOCENTRIC ORBIT OF THE EARTH-MOON
257 C BARYCENTER. (UNITLESS)
258 C 5. XL(2) - COEFFICIENTS FOR SUN'S MEAN LONGITUDE. (RAD,RAD/SEC)
259 C 6. XM(2) - COEFFICIENTS FOR EARTH-MOON BARYCENTER MEAN ANOMALY.
260 C (RAD,RAD/SEC)
261 C 7. D(2) - COEFFICIENTS FOR MEAN ELONGATION OF THE MOON FROM THE SUN.
262 C (RAD,RAD/SEC)
263 C 8. XLLJ(2) - COEFFICIENTS FOR MEAN ELONGATION OF THE SUN FROM JUPITER.
264 C (RAD,RAD/SEC)
265 C 9. XLLSA(2)- COEFFICIENTS FOR MEAN ELONGATION OF THE SUN FROM SATURN.
266 C (RAD,RAD/SEC)
267 C 10. XMJ(2) - COEFFICIENTS FOR JUPITER'S MEAN ANOMALY. (RAD,RAD/SEC)
268 C 11. XMSA(2) - COEFFICIENTS FOR SATURN'S MEAN ANOMALY. (RAD,RAD/SEC)
269 C 12. ACT(15) - AMPLITUDES OF TERMS IN CT-AT. (SEC)
270 C

272  COMMON / cmath / pi, twopi, halfpi, convd, convds, convhs, secday
273  EXTERNAL cmathb

275 C 1. SECDAY - THE CONVERSION FACTOR OF COORDINATE TIME SECONDS PER

277 C
278  include 'ccon.i'
279 C VARIABLES 'FROM':
280 C 1. KCTIC - THE CTIMG UTILITY ROUTINE FLOW CONTROL FLAG.
281 C 2. KCTID - THE CTIMG UTILITY ROUTINE DEBUG OUTPUT FLAG.
282 C
283 C 1.2.3 PROGRAM SPECIFICATIONS -
284 C
285  Real*8 AT, CFSITE(3,2), SITLON(2), UTC, XJD, CT, DATDCT, DLPGR
286 C
287 C 1.2.4 DATA BASE ACCESS -
288 C 'PUT' VARIABLES:
289 C 1. CT - THE COORDINATE TIME FRACTION OF THE COORDINATE TIME DAY
290 C AT OBSERVATION SITE #1. (DAYS)
291 C ACCESS CODES:
292 C 1. 'CT SITE1' - THE DATA BASE ACCESS CODE FOR THE COORDINATE TIME

294 C
295 C 1.2.5 EXTERNAL INPUT/OUTPUT - POSSIBLE DEBUG OUTPUT
296 C
297 C 1.2.6 SUBROUTINE INTERFACE -
298 C CALLER SUBROUTINES: DRIVG
299 C CALLED SUBROUTINES: DCOS, DSIN, PUT4
300 C
301 C 1.2.7 CONSTANTS USED - SECDAY, TWOPI, ATCTEP, D1950, ECCEN,
302 C XL, XM, D, XLLJ, XLLSA, XMJ, XMSA, ACT
303 C
304 C 1.2.8 PROGRAM VARIABLES -
305 C 1. XLE - SUN'S MEAN LONGITUDE. (RAD,RAD/SEC)
306 C 2. XME - EARTH-MOON BARYCENTER MEAN ANOMALY. (RAD,RAD/SEC)
307 C 3. DE - MEAN ELONGATION OF THE MOON FROM THE SUN. (RAD,RAD/SEC)
308 C 4. XLLJE - MEAN ELONGATION OF THE SUN FROM JUPITER. (RAD,RAD/SEC)
309 C 5. XLLSAE - MEAN ELONGATION OF THE SUN FROM SATURN. (RAD,RAD/SEC)
310 C 6. XMJE - JUPITER'S MEAN ANOMALY. (RAD,RAD/SEC)
311 C 7. XMSAE - SATURN'S MEAN ANOMALY. (RAD,RAD/SEC)
312 C 8. ANOMEC - THE ECCENTRIC ANOMALY OF THE HELIOCENTRIC ORBIT OF THE
313 C EARTH-MOON BARYCENTER. (RAD)
314 C 9. SECEP - THE NUMBER OF SECONDS ELAPSED SINCE THE EPOCH JAN 1, 1950
315 C 0HR UT.
316 C 10. UTL - UTC + EAST LONGITUDE OF STATION 1. (RAD)
317 C 11. U - STATION 1 DISTANCE FROM SPIN AXIS. (KM)
318 C 12. V - STATION 1 DISTANCE ABOVE EQUATOR. (KM)
319 C 13. CTP - PERIODIC TERMS IN CT. (SEC)
320 C
321 C 1.2.9 PROGRAMMER - DALE MARKHAM 01/17/77
322 C PETER DENATALE 07/18/77
323 C BRUCE SCHUPLER 02/21/78
324 C CHOPO MA 08/06/81
325 C Jim Ryan 89.07.26 Documentation simplified.
326 C Gregg Cooke 90.02.14 Changed name to avoid conflict
327 C with hp-ux libraries.
328 C
329  Real*8 XLE, XME, DE, XLLJE, XLLSAE, XMJE, XMSAE, ANOMEC,
330  . secep, utl, u, v, ctp
331 C
332 C CTIMG Program Structure
333 C
334 C Compute the number of seconds elapsed since epoch Jan 1, 1950
335 C 0 hours UT.
336  secep = ( xjd + at - d1950 ) * secday
337 C
338 C Compute the astronomical arguments.
339 C
340 C Compute the mean anomaly of the heliocentric orbit of the
341 C Earth-Moon barycenter. Moyer Eq. 44
342  xme = xm(1) + xm(2) * secep
343 C
344 C Compute the eccentric anomaly of the heliocentric orbit of the
345 C Earth_moon barycenter. Moyer Eq. 40
346  anomec = xme + eccen * dsin( xme )
347 C
348 C Compute the sun's mean longitude. Moyer Eq. 43
349  xle = xl(1) + xl(2) * secep
350 C
351 C Compute mean elongation of the Moon from the Sun. Moyer Eq. 45
352  de = d(1) + d(2) * secep
353 C
354 C Compute mean elongation of the Sun from Jupiter. Moyer Eq. 46
355  xllje = xllj(1) + xllj(2) * secep
356 C
357 C Compute mean elongation of the Sun from Saturn. Moyer Eq. 47
358  xllsae = xllsa(1) + xllsa(2) * secep
359 C
360 C Compute Jupiter's mean anomaly - approximation for eccentric anomaly
361 C Moyer Eq. 48
362  xmje = xmj(1) + xmj(2) * secep
363 C
364 C Compute Saturn's mean anomaly - approximation for eccentric anomaly
365 C Moyer Eq. 49
366  xmsae = xmsa(1) + xmsa(2) * secep
367 C
368 C Compute periodic contributions to coordinate time. The Moyer
369 C Equation (No. 38) calls for UT1+LAMDA,i.e., UT1 plus the crust
370 C fixed site longitude in the following 'UTL' is that quantity.
371  utl = utc * twopi + sitlon(1)
372 C
373 C Compute the station spin axis distance and Z-component in kilometers.
374  u = dsqrt( cfsite(1,1)**2 + cfsite(2,1)**2 ) / 1.d3
375  v = cfsite(3,1) / 1.d3
376 C
377 C Now evaluate Moyer's Equation # 43 with delta T sub A removed.
378  ctp = act(1) * dsin( anomec ) + act(2) * dsin( de )
379  . + act(3) * u * dsin( utl )
380  . + act(4) * u * dsin( utl - xme )
381  . + act(5) * u * dsin( utl - 2.d0*xme )
382  . + act(6) * u * dsin( utl + 2.d0*xle )
383  . + act(7) * u * dsin( utl + 2.d0*xle + xme )
384  . + act(8) * u * dsin( utl - de ) + act(9) * v * dcos( xle )
385  . + act(10) * dsin( xmje ) + act(11) * dsin( xmsae )
386  . + act(12) * dsin( xllje ) + act(13) * dsin( xllsae )
387  . + act(14) * u * dsin( utl + xllje )
388  . + act(15) * u * dsin( utl + xllsae )
389 C
390 C Compute coordinate time depending on flow control.
391  IF (kctic.EQ.0) ct = at + (atctep + ctp) / secday
392  IF (kctic.EQ.1) ct = at
393  IF (kctic.EQ.2) ct = at + atctep / secday
394 C
395 C Compute the partial derivative of the long period term in
396 C 'AT minus CT' offset with respect to the coordinate time.
397 C The long period terms are those without any diurnal signature,
398 C specifically those withut UTL. The leading (-) is required
399 C since the series above defines the long period terms for
400 C 'CT minus AT'.
401 C
402  dlpgr = - ( act(1) * dcos( anomec )
403  1 * ( 1.d0 + eccen * dcos( xme ) ) * xm(2)
404  2 + act(2) * dcos( de ) * d(2)
405  3 - act(9) * v * dsin( xle ) * xl(2)
406  4 + act(10) * dcos( xmje ) * xmj(2)
407  5 + act(11) * dcos( xmsae ) * xmsa(2)
408  6 + act(12) * dcos( xllje ) * xllj(2)
409  7 + act(13) * dcos( xllsae ) * xllsa(2) )
410 C
411  IF( kctic.NE.0) dlpgr = 0.d0
412 C
413 C Compute the partial derivative of the atomic time with respect to
414 C the coordinate time. The (+) reverses the sign of DLPGR.
415  datdct = 1.d0 + dlpgr
416 C
417 C PUT the coordinate time fracton of the coordinate time day into
418 C the database.
419  CALL put4 ('CT SITE1 ', ct, 1, 1, 1 )
420 C
421 C Check KCTID for debug output.
422  IF ( kctid .EQ. 0 ) GO TO 1100
423  WRITE ( 6, 9)
424  9 FORMAT(1x, "Debug output for subroutine CTIME." )
425  WRITE(6,8)' XL ',xl
426  8 FORMAT(a,4d25.16/(7x,5d25.16))
427  WRITE(6,8)' XLE ',xle
428  WRITE(6,8)' XM ',xm
429  WRITE(6,8)' XME ',xme
430  WRITE(6,8)' D ',d
431  WRITE(6,8)' DE ',de
432  WRITE(6,8)' XLLJ ',xllj
433  WRITE(6,8)' XLLJE ',xllje
434  WRITE(6,8)' XLLSA ',xllsa
435  WRITE(6,8)' XLLSAE ',xllsae
436  WRITE(6,8)' XMJ ',xmj
437  WRITE(6,8)' XMJE ',xmje
438  WRITE(6,8)' XMSA ',xmsa
439  WRITE(6,8)' XMSAE ',xmsae
440  WRITE(6,8)' ECCEN ',eccen
441  WRITE(6,8)' ACT ',act
442  WRITE(6,8)' ANOMEC ',anomec
443  WRITE(6,8)' CTP ',ctp
444  WRITE(6,8)' D1950 ',d1950
445  WRITE(6,8)' SECDAY ',secday
446  WRITE(6,8)' SECEP ',secep
447  WRITE(6,8)' UTL ',utl

449  WRITE(6,8)' V ',v

451  WRITE(6,8)' ATCTEP ',atctep
452  WRITE ( 6, 9200 ) at, cfsite(1,1), cfsite(2,1), sitlon(1),
453  . utc, xjd, ct, datdct, dlpgr
454  9200 FORMAT (1x, "AT = ", d30.16, /, 1x,
455  . "CFSITE(1,1) = ", d30.16, /, 1x,
456  . "CFSITE(2,1) = ", d30.16, /, 1x,
457  . "SITLON (1) = ", d30.16, /, 1x,
458  . "UTC = ", d30.16, /, 1x,
459  1 "XJD = ", d30.16, /, 1x,
460  2 "CT = ", d30.16, /, 1x,
461  3 "DATDCT = ", d30.16, /, 1x,
462  4 "DLPGR = ", d30.16 )
