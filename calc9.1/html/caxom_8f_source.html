490 C their time derivatives and the partial derivatives of the component of
491 C the antenna axis offsets in the apparent direction of the source and the
492 C time derivative of that component with respect to the antenna axis
493 C offsets at each site.
494 C
495  DO 800 l = 1,2
496  ll = l
497 C
498 C Check for geocenter station
499  IF (l. eq. nzero) Go to 750
500 C
501 C Identify the unit vector representing the antenna fixed axis in a
502 C topocentric reference system. The topocentric reference system sits on the
503 C Earth's surface at the observation site with axes pointing 1) radially Up
504 C (X-AXIS), 2) East (Y-axis), and 3) North (Z-axis).
505 C
506 C Identify the antenna axis type in order to obtain the correct algorithm:
507 C
508 C Equatorial mount - fixed axis points at the North Celestial Pole.
509  If (kaxis(l) .EQ. 1) Then
510  tcaxis(1) = dsin( sitlat(l) )
511  tcaxis(2) = 0.d0
512  tcaxis(3) = dcos( sitlat(l) )
513  GO TO 310
514  Endif
515 C
516 C X-Y mount - antenna fixed axis points North-South. (i.e. the antenna fixed
517 C axis is in the plane of the horizon pointed North-South.)
518  If (kaxis(l) .EQ. 2) Then
519  tcaxis(1) = 0.d0
520  tcaxis(2) = 0.d0
521  tcaxis(3) = 1.d0
522  GO TO 310
523  Endif
524 C
525 C Alt-Az mount - the antenna fixed axis points at the zenith.
526  If (kaxis(l) .EQ. 3) Then
527  tcaxis(1) = 1.d0
528  tcaxis(2) = 0.d0
529  tcaxis(3) = 0.d0
530  GO TO 310
531  Endif
532 C
533 C X-Y mount - antenna fixed axis pointing East-West. (i.e. The antenna
534 C fixed axis is in the plane of the horizon pointed East-West.)
535  If (kaxis(l) .EQ. 4) Then
536  tcaxis(1) = 0.d0
537  tcaxis(2) = 1.d0
538  tcaxis(3) = 0.d0
539  GO TO 310
540  Endif
541 C
542 C Old Richmond antenna system - axes oriented as for an equatorial mount at
543 C latitude +39.06 degrees and rotated 0.12 degrees West of North.
544  If (kaxis(l) .EQ. 5) Then
545  zc = dcos(richm(1)*convd)
546  tcaxis(1) = dsin(richm(1)*convd)
547  tcaxis(2) = -zc*dsin(richm(2)*convd)
548  tcaxis(3) = zc*dcos(richm(2)*convd)
549  GO TO 310
550  Endif
551 C
552 C Antenna axis type invalid, a message is written and the program stops.
553  IF ((kaxis(l) .le. 0 ) .OR. (kaxis(l) .gt. 5)) Then
554  WRITE ( 6, 9300 ) ll, kaxis(ll)
555  CALL ckill (6haxog , 0, 0)
556  Endif
557  9300 FORMAT (1x, "THE PROGRAM HAS TERMINATED IN SUBROUTINE AXOG. ",
558  1 'KAXIS (', i2, ') = ', i2, '.' )
559 C
560 C******************************************************************************
561 C Rotate the J2000.0 source unit vector to the topocentric reference system.
562 C
563 C Compute the rotation matrix which rotates from the geocentric crust
564 C fixed reference system to the topocentric reference system.
565  310 CALL mtran ( tctocf(1,1,l), cftotc )
566 C
567 C Compute the rotation matrix which rotates from the J2000.0
568 C reference system to the geocentric crust fixed system.
569  CALL mtran ( r2000(1,1,1), tr2000(1,1,1) )
570 C
571 C Rotate the aberrated J2000.0 source unit vector to the crust fixed
572 C system and then to the topocentric system.
573  CALL vecrt ( tr2000(1,1,1),star_aberrated(1,l),cfstar(1,1))
574  CALL vecrt ( cftotc, cfstar(1,1), tcstar(1,1) )
575 C
576 C*****************************************************************************
577 C Correct the aberrated topocentric source unit vector for atmospheric
578 C refraction.
579 C
580 C Compute the zenith angle of the aberrated source.
581  zen = halfpi - elev(l,1)
582 C Compute the azimuth of the aberrated source.
583  azmuth = az(l,1)
584 C Compute values needed for function sbend.
585  el_rad = elev(l,1)
586  sithit = sitheight(l)
587  temp_k = 293.15 - (6.5d-3)*sithit
588  x = 1.d0 - (6.5d-3)*sithit/293.15d0
589  press_hg = 760.d0 * (x**5.26d0)
590  humid_f = .5d0
591 C Compute atmospheric bending.
592  zencor = sbend(el_rad,temp_k,humid_f,press_hg)
593 C
594 C *************************************************************************
595 C Compute the corrected (aberrated/refracted) topocentric star unit
596 C vector. (NOTE: This corrects the vector so that it is pointing nearer
597 C to the zenith).
598  ctcstr(1) = dcos( zen - zencor )
599  ctcstr(2) = dsin( zen - zencor ) * dsin( azmuth )
600  ctcstr(3) = dsin( zen - zencor ) * dcos( azmuth )
601 C Make sure we still have a unit vector
602  amag = vecmg(ctcstr)
603  do kx = 1,3
604  ctcstr(kx) = ctcstr(kx)/amag
605  enddo
606 C
607 C Compute the CT time derivative of the aberrated source unit vector
608 C in the topocentric reference system.
609  CALL mtran ( r2000(1,1,2), tr2000(1,1,2) )
610  CALL vecrt ( tr2000(1,1,2),star_aberrated(1,l),cfstar(1,2))
611  CALL vecrt ( cftotc, cfstar(1,2), tcstar(1,2) )
612 C
613 C .............................................................................
614 C New code for computing the three dimensional components of the axis offset
615 C vector. D. Gordon 93MAY13 thru 93JUL27
616 C
617 C Compute topocentric axis offset vector (change in effective site position
618 C due to axis offset orientation) and its time derivative:
619 C
620 C DELAY COMPUTATION:
621 C ------------------
622 C Take triple vector product to get direction of axis offset vector:
623  call crosp(tcaxis,ctcstr,vectr)
624  call crosp(vectr,tcaxis,vec1)
625 C Compute magnitude of vector so we can normalize it
626  absvec1 = vecmg(vec1)
627 C Normalize to get topocentric unit axis offset vector
628  do kx=1,3
629  axistc(kx,l) = vec1(kx) / absvec1
630  enddo
631 C
632 C Rotate axis offset vector to the crust fixed frame
633  call vecrt(tctocf(1,1,l),axistc(1,l),axiscf(1,l))
634 C Then rotate axis offset vector to the J2000 frame
635  call vecrt(r2000(1,1,1),axiscf(1,l),uaxis2000(1,l))
636 C
637 C Derivative of CTCSTR: (aberrated/refracted topocentric source unit vector)
638 C --------------------
639 C Rotate atmosphere corrected aberrated source unit vector to the
640 C crust fixed frame:
641  call vecrt(tctocf(1,1,l),ctcstr,ccfstr)
642 C Then rotate to the J2000 frame:
643  call vecrt(r2000(1,1,1),ccfstr,c2000str)
644 C
645 C Take time derivative and rotate back to crust fixed frame
646  call vecrt(tr2000(1,1,2),c2000str,dccfstr)
647 C
648 C Rotate source vector time derivative back to topocentric frame
649  call vecrt(cftotc,dccfstr,dctcstr)
650 C
651 C RATE COMPUTATION:
652 C -----------------
653 C Take derivative of triple vector product (TCAXIS is a constant)
654  call crosp(tcaxis,dctcstr,vectr)
655  call crosp(vectr,tcaxis,dvec1)
656 C Derivative of magnitude of vec1
657  dabsvec1 = dotp(vec1,dvec1) / absvec1
658 C Compute derivative of unit axis offset vector in topocentric frame
659  do kx=1,3
660  daxistc(kx,l) = dvec1(kx)/absvec1 -
661  . vec1(kx) * dabsvec1 / absvec1**2
662  enddo
663 C
664 C Rotate axis offset time derivative to crust fixed frame
665  call vecrt(tctocf(1,1,l),daxistc(1,l),daxiscf(1,l))
666 C
667 C Then rotate to J2000 frame:
668  call vecrt(r2000(1,1,1),daxiscf(1,l),vec2)
669  call vecrt(r2000(1,1,2), axiscf(1,l),vec3)
670  call vecad(vec2,vec3,duaxis2000(1,l))
671 C
672 C Compute partial derivative of the components of the antenna axis offset
673 C in the apparent direction of the source with respect to the antenna axis
674 C offset.
675  dcomp(l,1) = dotp(c2000str,uaxis2000(1,l))
676  dcomp(l,2) = dotp(c2000str,duaxis2000(1,l))
677 C
678 C Convert unit axis offset vector and its derivative to actual axis offset
679 C vector and its derivative.
680  do kx=1,3
681  axis2000(kx,l) = uaxis2000(kx,l) * axoff(l)
682  daxis2000(kx,l) = duaxis2000(kx,l) * axoff(l)
683  enddo
684 C
685 C For use later in AXOP, save the topocentric UP (change in height) component
686 C of the unit axis offset vector and its time derivative
687  udel_h(l,1) = axistc(1,l) ! unit UP component
688  udel_h(l,2) = daxistc(1,l) ! unit UP time derivative
689 C
690 C .............................................................................
691 C Now compute new axis offset partial with respect to AXOFF (3-part formula,
692 C see D. Gordon reference)
693 C
694  sinez = dsin(elev(l,1)+zencor)
695  dsinez = dcos(elev(l,1)+zencor) * elev(l,2)
696 C
697 C Vacuum part 1:
698 C! term1 = -Dotp( uaxis2000(1,L),Star)
699 C! dterm1 = -Dotp(duaxis2000(1,L),Star)
700 C Vacuum part 1 with aberration correction:
701  term1 = -dotp( uaxis2000(1,l),star_aberrated(1,l))
702  dterm1 = -dotp(duaxis2000(1,l),star_aberrated(1,l))
703 C
704 C Vacuum part 2 with aberration correction and refraction/geometric part:
705 c DotDot1 = Dotp(Star,c2000str) - 1.D0
706 C dDotDot1 = 0.D0 ! approximation
707  dotdot1 = dotp(star_aberrated(1,l),c2000str) - 1.d0
708  term2 = axistc(l,1) * dotdot1 / sinez
709  dterm2 = daxistc(l,1) * dotdot1 / sinez -
710  * axistc(l,1) * dotdot1 * dsinez / sinez**2
711 C
712 C Now combine everything to get the first two parts of the new axis
713 C offset delay and rate partials with respect to AXOFF. (The last part, the
714 C additional atmosphere correction to the reference height is done in AXOP.)
715  dcomp_new(l,1) = (term1 + term2) / vlight
716  dcomp_new(l,2) = (dterm1 + dterm2) / vlight
717 C******************************************************************************
718 C
719 C Now do feedbox rotation computations
720 C
721  IF(kpanc.EQ.1) THEN
722  pangl(l) = 0.0d0
723  feed_cor(l,1) = 0.0d0
724  feed_cor(l,2) = 0.0d0
725  GO TO 156
726  ENDIF
727 C
728 C Define vector TNCP(3), a unit vector in the direction of the North
729 C Celestial pole.
730  tncp(1) = dsin(sitlat(l))
731  tncp(2) = 0.d0
732  tncp(3) = dcos(sitlat(l))
733 C
734  za = zen - zencor
735  dza = -elev(l,2)
736  dazm = az(l,2)
737  If (kaxis(l) .NE. 1) Then
738  CALL fbox (tncp, tcaxis, ctcstr, azmuth, dazm, za, dza,
739  * rangl, drangl)
740  Else
741 C Equatorial Mount, no feedbox rotation
742  rangl = 0.0d0
743  drangl = 0.0d0
744  Endif
745 C
746 C Convert the feedbox rotation angle to degrees for this site.
747  pangl(l) = rangl/convd
748 C Compute the feedbox rotation corrections for group delay and phase delay
749 C rate for this site. The units are seconds and sec/sec.
750  feed_cor(l,1) = 0.d0
751  feed_cor(l,2) = drangl/twopi/ref_freq
752 C
753 C To correct the phases, either ADD the quantity [PANGL(2) - PANGL(1)]
754 C to the total or residual observed phases, or SUBTRACT it from the
755 C theoretical (model) phases.
756 C Note that the group delay corrections are always zero.
757 C To correct the phase delay rates, either ADD the quantity
758 C [FEED_COR(2,2) - FEEDCOR(1,2)] to the observed rates, or SUBTRACT
759 C it from the theoretical (model) rates.
760 C
761  156 CONTINUE
762 C******************************************************************************
763 C See if debug is needed.
764  IF ( kaxod .ne. 0 ) Then
765 C
766  If (l.eq.1) then
767  WRITE ( 6, 9100 )
768  endif
769 C
770  If (l.eq.2) then
771  WRITE ( 6, 9101 )
772  WRITE ( 6, 9200 ) kaxis, r2000, sitlat, star, tctocf
773  WRITE(6,8)' TR2000 ',tr2000
774  WRITE(6,8)' STAR_ABERRATED ',star_aberrated
775  WRITE(6,8)' AXOFF ', axoff
776  WRITE(6,8)' PANGL ', pangl
777  WRITE(6,8)' FEED_COR ', feed_cor
778  endif
779 C
780  9100 FORMAT (20x, "Debug output for subroutine AXOG." )
781  9200 FORMAT (1x, "KAXIS = ", 2 ( i2, 10x ), /, 1x,
782  1 "R2000 = ", 9 ( 3 ( d30.16, 10x ), /, 1x ), /, 1x,
783  2 "SITLAT = ", 2 ( d30.16, 10x ), /, 1x,
784  3 "STAR = ", 3 ( d30.16, 4x ), /, 1x,
785  4 "TCTOCF = ", 6 ( 3 ( d30.16, 10x ), /, 1x ) )
786  9101 FORMAT (1x, "Dump for both stations: " )
787  8 FORMAT(a,3d25.16,/,5(7x,3d25.16))
788 C
789  write(6,'(/,10x," ******** DUMP FOR SITE",i2," ******** ")') l
790  WRITE(6,8)' TCAXIS ',tcaxis
791  WRITE(6,8)' CFTOTC ',cftotc
792  WRITE(6,8)' CFSTAR ',cfstar
793  WRITE(6,8)' TCSTAR ',tcstar
794  WRITE(6,8)' SURREF ',surref
795  WRITE(6,8)' ZEN ',zen
796  WRITE(6,8)' TANEL ',tanel
797  WRITE(6,8)' AZMUTH ',azmuth
798  WRITE(6,8)' ZENCOR ',zencor
799  write(6,'("vec1: ",3D23.14)') vec1

801  write(6,'("absvec1: ",3D23.14)') absvec1
802  write(6,'("dabsvec1: ",3D23.14)') dabsvec1
803  write(6,'("axistc: ",3D23.14)') axistc(1,l),axistc(2,l)
804  * ,axistc(3,l)
805  write(6,'("daxistc: ",3D23.14)') daxistc(1,l), daxistc(2,l),
806  * daxistc(3,l)
807  write(6,'("axiscf: ",3D23.14)') axiscf(1,l),axiscf(2,l)

810  * ,daxiscf(3,l)

812  write(6,'("ccfstr: ",3D23.14)') ccfstr
813  write(6,'("dccfstr: ",3D23.14)') dccfstr

830 C
831  ENDIF
832  Go to 800
833 C
834  750 CONTINUE

839  enddo

842  dcomp_new(l,1) = 0.0d0

849  feed_cor(l,2) = 0.0d0
850 C
851 C Close loop running over the observing sites.
852  800 CONTINUE

1344  SUBROUTINE fbox ( TNCP, TCAXIS, CTCSTR, AZ, DAZ, ZA, DZA,
1345  * RANGL, DRANGL )
1346 C
1347 C FBOX computes the feedbox rotation angle, relative to the source.
1348 C It computes the angle, looking up at the sky and measured at the
1349 C source, between the North Celestial Pole and the antenna fixed axis.
1350 C An increase in the feedbox rotation angle means a clockwise
1351 C rotation of the feedbox, for an observer looking up at the sky.
1352 C The angle itself is arbitrary. The important quantity is the
1353 C difference in the feedbox rotations at the two antennas, and how it
1354 C changes from one observation to the next.
1355 C
1356 C It's not clear what to do with these angle in all case. For RCP data
1357 C in geodesy, we correct the phase by adding the remote station
1358 C correction and subtracting the reference station correction.
1359 C
1360 C The algorithm makes two coordinate rotations to put the star vector
1361 C along the X-axis. The azimuths of the NCP and the fixed axis in the
1362 C transformed Y-Z plane are then easily found and differenced.
1363 C
1364 C This subroutine replaces subroutine PANG in Calc 9.0, and will handle
1365 C all azis types.
1366 C
1367 C Input Variables:
1368 C 1) CTCSTR(3) - Topocentric unit vector in the direction of the
1369 C source. [X = Up = COS(Zenith Angle). Y = East =
1370 C SIN(Zenith Angle) * SIN(Azimuth). Z = North =
1371 C SIN(Zenith Angle) * COS(Azimuth). For greatest
1372 C accuracy, the effects of atmospheric refraction
1373 C and aberration should be included.]
1374 C 2) TCAXIS(3) - Topocentric unit vector in the direction of the
1375 C antenna's fixed axis. [Altazimuth: X = 1.0,
1376 C Y = 0.0, Z = 0.0. X/Y North (Gilcreek):
1377 C X = 0., Y = 0., Z = 1.0. X/Y East (HARTRAO):
1378 C X = 0., Y = 1.0., Z = 0.]
1379 C 3) TNCP(3) - Topocentric unit vector in the direction of the
1380 C North Celestial Pole. [X = SIN(Geodetic Latitude),
1381 C Y = 0.0, Z = COS(Geodetic Latitude)]
1382 C 4) AZ - Azimuth angle of source, in radians.
1383 C 5) DAZ - Time derivative of Azimuth angle. (radians/sec)
1384 C Zeros can be used here if the user does not care
1385 C about the rate of change of the rotation angle, or
1386 C about the correction to the phase delay rate.
1387 C 6) ZA - Zenith angle of source, in radians. Include
1388 C refraction and aberration for greatest accuracy.
1389 C 7) DZA - Time derivative of zenith angle. (radians/sec)
1390 C Zeros can be used here if the user does not care
1391 C about the rate of change of the rotation angle, or
1392 C about the correction to the phase delay rate.
1393 C Output Variables:
1394 C 1) RANGL - The clockwise rotation angle of the feedbox. (rad)
1395 C 2) DRANGL - The time derivative of RANGL. (rad/sec)
1396 C
1397 C
1398  Real*8 PI,TWOPI,HALFPI,CONVD,CONVDS,CONVHS,SECDAY
1399  COMMON /cmath/ pi,twopi,halfpi,convd,convds,convhs,secday
1400  EXTERNAL cmathb
1401 C VARIABLES 'FROM':
1402 C 1. HALFPI - THE VALUE OF PI/2
1403 C 2. CONVD - THE CONVERSION FACTOR FROM DEGREES TO RADIANS
1404 C (RAD/DEG)
1405 C
1406 C Program History:
1407 C D. Gordon 98.11.10 Subroutine written at GSFC. Replaces subroutine
1408 C PANG.
1409 C
1410 C
1411  Real*8 TNCP(3), TCAXIS(3), CTCSTR(3), AZ, DAZ,
1412  * za, dza, rangl, drangl
1413  REAL*8 P1AZ(3,3), P2ZA(3,3), P12(3,3), DP1AZ(3,3), DP2ZA(3,3),
1414  * dp12(3,3), tt1(3,3), tt2(3,3)
1415  REAL*8 NCP1(3), AXIS1(3), STAR1(3), DNCP1(3), DAXIS1(3)
1416  REAL*8 AZNCP, AZAXIS, DAZNCP, DAZAXIS
1417 C
1418 C Construct rotation matrix to rotate about the X-axis by the angle -AZ
1419 C and its derivative.
1420  CALL rotat (-az, 1, p1az)
1421  CALL drott (-az, -daz, 1, dp1az)
1422 C Construct rotation matrix to rotate about the Y-axis by the angle -ZA
1423 C and its derivative.

1425  CALL drott (-za, -dza, 2, dp2za)
1426 C Combine the above rotation matrices and the derivative.
1427  CALL mmul2 (p2za, p1az, p12)
1428  CALL mmul2 (dp2za, p1az, tt1)
