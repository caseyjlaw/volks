18 C LOADING MODULE FLOW CONTROL MESSAGE.
19 C 4. 'OCE DELD' - THE DATA BASE ACCESS CODE FOR THE SITE
20 C DEPENDENT OCEAN LOADING DISPLACEMENTS AND

48 C 1.3 OCEA PROGRAM STRUCTURE
49 C
50 C Add for ocean loading module text message.
51  CALL adda (1,'OCE MESS','Ocean loading message definition',
52  1 40, 1, 1 )
53 C
54 C Add for module flow control message.
55  CALL adda (1,'OCE CFLG','Ocean load flow control mess def',
56  1 40, 1, 1 )
57 C
58 C Add for module contributions array. This is the old CALC-6 contribution,
59 C that is, one site-independent delay and rate for all ocean loading effects.
60  CALL addr (2,'OCE CONT','Ocean loading contributions def.',
61  1 2, 1, 1 )
62 C
63 C Add for site depending displacement and velocity array.
64  CALL addr (2,'OCE DELD','Ocean load site depndnt displace',
65  . 3, 2, 2 )
66 C
67 C Add for site dependent contribution to the delay and rate for the horizontal
68 C corrections.
69  CALL addr (2,'OCE HORZ','Site-dep ocean cont - horizontal',
70  . 2, 2, 1 )
71 C
72 C Add for site dependent contribution to the delay and rate for the vertical
73 C corrections.
74  CALL addr (2,'OCE VERT','Site-dep ocean cont - vertical ',
75  . 2, 2, 1 )
76 C
77  RETURN
78  END

384 C OCEG PROGRAM STRUCTURE
385 C
386 C The ocean loading geometry for sites #1 and #2 are calculated separately by
387 C running a loop over the two sites.
388 C
389 C Compute the time dependent angular momentum for the 11 tidal arguments.
390  IF (tskip.ne.1)
391  * CALL ocarg ( ut1, xjd, angle )
392 C
393 C Loop twice for the calculation of the height displacement due to ocean
394 C loading at sites #1 and 2.
395 C
396  DO l = 1, 2
397 C
398 C Check for Geocenter site
399  IF (l .eq. nzero) Then
400  Do k=1,3
401  deltao(k,l) = 0.d0
402  deltav(k,l) = 0.d0
403  tcdis(k,l) = 0.d0
404  tcvel(k,l) = 0.d0
405  zloadp(k,l) = 0.d0
406  zloadp_hor(k,l) = 0.d0
407  zloadp_ver(k,l) = 0.d0
408  zloadv(k,l) = 0.d0
409  zloadv_hor(k,l) = 0.d0
410  zloadv_ver(k,l) = 0.d0
411  Enddo
412  Go to 150
413  ENDIF
414 C
415 C Compute the topocentric displacement DELTAO and DELTAV for each site due to
416 C ocean loading (for the 11 main tides in 3 directions).
417 C
418  DO k = 1, 3
419  deltao(k,l) = 0.d0
420  deltav(k,l) = 0.d0
421  DO j = 1,11
422  deltao(k,l) = oceamp(j,k,l) * dcos(angle(j)
423  . - ocephs(j,k,l)) + deltao(k,l)
424  deltav(k,l) = - oceamp(j,k,l) * dsin(angle(j)
425  . - ocephs(j,k,l)) * speed(j)
426  . + deltav(k,l)
427  ENDDO
428  ENDDO
429 C
430 C Change the signs on the horizontal displacements and velocities.
431 C (See explanation in the subroutine history above.)
432 C
433  deltao(2,l) = -deltao(2,l)
434  deltav(2,l) = -deltav(2,l)
435  deltao(3,l) = -deltao(3,l)
436  deltav(3,l) = -deltav(3,l)
437 C
438 C Load the topocentric ocean loading displacement vector. Also load up vector
439 C that contains only the vertical and only the horizontal effects.
440  tcdis(1,l) = deltao(1,l)
441  tcdis(2,l) = deltao(2,l)
442  tcdis(3,l) = deltao(3,l)
443 C
444  tcdis_hor(1,l) = 0.d0
445  tcdis_hor(2,l) = deltao(2,l)
446  tcdis_hor(3,l) = deltao(3,l)
447 C
448  tcdis_ver(1,l) = deltao(1,l)
449  tcdis_ver(2,l) = 0.d0
450  tcdis_ver(3,l) = 0.d0
451 C
452 C Load the topocentric ocean loading velocity vector. As above load like
453 C quantities for horizontal and vertical.
454  tcvel(1,l) = deltav(1,l)
455  tcvel(2,l) = deltav(2,l)
456  tcvel(3,l) = deltav(3,l)
457 C
458  tcvel_hor(1,l) = 0.d0
459  tcvel_hor(2,l) = deltav(2,l)
460  tcvel_hor(3,l) = deltav(3,l)
461 C
462  tcvel_ver(1,l) = deltav(1,l)
463  tcvel_ver(2,l) = 0.d0
464  tcvel_ver(3,l) = 0.d0
465 C
466 C Rotate the displacements and velocities to the crust fixed geocentric
467 C coordinate system.
468  CALL vecrt(tctocf(1,1,l),tcdis(1,l),cfdis(1,l))
469  CALL vecrt(tctocf(1,1,l),tcvel(1,l),cfvel(1,l))
470 C
471  CALL vecrt(tctocf(1,1,l),tcdis_hor(1,l),cfdis_hor(1,l))
472  CALL vecrt(tctocf(1,1,l),tcvel_hor(1,l),cfvel_hor(1,l))
473 C
474  CALL vecrt(tctocf(1,1,l),tcdis_ver(1,l),cfdis_ver(1,l))
475  CALL vecrt(tctocf(1,1,l),tcvel_ver(1,l),cfvel_ver(1,l))
476 C
477 C Rotate the crust fixed geocentric displacements to J2000.0.
478  CALL vecrt(r2000(1,1,1),cfdis(1,l) ,zloadp(1,l) )
479  CALL vecrt(r2000(1,1,1),cfdis_hor(1,l),zloadp_hor(1,l))
480  CALL vecrt(r2000(1,1,1),cfdis_ver(1,l),zloadp_ver(1,l))
481 C
482 C Rotate the crust fixed geocentric velocities to J2000.0.
483 C
484 C Compute the contribution to the J2000.0 velocities due to the effect of
485 C changed site positon due to ocean loading.
486  CALL vecrt(r2000(1,1,2),cfdis(1,l),zloav1(1,l))
487  CALL vecrt(r2000(1,1,2),cfdis_hor(1,l),zloav1_hor(1,l))
488  CALL vecrt(r2000(1,1,2),cfdis_ver(1,l),zloav1_ver(1,l))
489 C
490 C Compute the contribution to the J2000.0 velocities due to the effect of the
491 C changing ocean loading displacement.
492  CALL vecrt(r2000(1,1,1),cfvel(1,l),zloav2(1,l))

501  150 CONTINUE
502 C
503  ENDDO

505 C Check KOCEC to determine if the ocean loading module is to be turned off.
506 C
507 C Handle the normal case (Contributions only)
508  IF (kocec .EQ. 0 .or. kocec.eq.3) THEN
509  DO l = 1,2
510  DO i = 1,3
511  xloadp(i,l) = 0.d0
512  xloadv(i,l) = 0.d0
513  ENDDO
514  ENDDO
515  ENDIF
516 C
517 C Handle the ocean loading module off case.
518  IF (kocec .EQ. 1) THEN
519  DO l = 1,2
520  DO i = 1,3
521  zloadp(i,l) = 0.d0
522  zloadp_hor(i,l) = 0.d0
523  zloadp_ver(i,l) = 0.d0
524  xloadp(i,l) = 0.d0
525

527  zloadv_hor(i,l) = 0.d0
528  zloadv_ver(i,l) = 0.d0
529  xloadv(i,l) = 0.d0
530  ENDDO
531  ENDDO
532  ENDIF
533 C
534 C Handle the special case (KOCEC=2) where the effect of ocean loading is
535 C added to the theoretical.
536  IF (kocec.EQ.2 .or. (calc_user.eq.'C' .and. apply_ocean.eq.'Y'))

538 C print *, 'Ocean loading being applied'

540  DO i = 1,3
541  xloadp(i,l) = zloadp(i,l)

543  ENDDO
544  ENDDO

553  xhold(i,1,l) = tcdis(i,l)
554  xhold(i,2,l) = tcvel(i,l)
555  ENDDO
556  ENDDO

560 C Check KOCED to determine if debug output is requested.
561  400 IF( koced .EQ. 0) GO TO 500
562  WRITE (6,9100)

879 C
880 C 1.2.8 PROGRAM VARIABLES
881 C 1. CAPT - THE NUMBER OF JULIAN CENTURIES BETWEEN JAN 0.5, 1900
882 C AND THE OBSERVATION.
883 C 2. FDAY - FRACTIONAL PART OF UNIVERSAL TIME (UT1) DAY IN

888 C (RAD)

890 C 7. JCAPD - Previous value of ICAPD. Checked to see if we can
891 C reuse the previous values of CAPT, H0, P0, and S0.
892 C
893 C 1.2.9 PROGRAMMER - CLYDE GOAD
894 C 83:10:08 HAROLD M. SCHUH
895 C 89:10:08 Jim Ryan ANGFAC and SPEED moved to ema.
896 C 91:11:04 Jim Ryan ANGFAC(8,3) changed from 0.0 to 1.0 to make
897 C consistent with IERS standards (Nov. 89)
898 C David Gordon 94.04.18 Converted to Implicit None.

900 C it to an integer - error had no effect. Put SPEED(11)

902 C replace DANGL(11)].
903 C David Gordon 94.09.22 Added JCAPD and setting it to the previous
904 C value of ICAPD. If equal to current ICAPD, we skip
905 C computation of CAPT, H0, S0, and P0. CAPT, H0, S0, P0,
906 C and JCAPD put in save block.

908 C 1.3 OCARG PROGRAM STRUCTURE
909 C

914  icapd = xjd - xjd75
915 C
916 C Check to see if we've started a new day. If so, we need to compute new values
917 C for CAPT, H0, S0, & P0. If not, then we can reuse the previous values.
918  If (icapd .ne. jcapd) Then
919 C
920  jcapd = icapd !Reset JCAPD for next observation
921 C
922 C Compute CAPT - Julian centuries since JAN 0.5, 1900.
923  capt = ( 27392.500528d0 + 1.000000035d0 * icapd ) / centj
924 C
925 C Compute mean longitude of sun at beginning of day.
926  h0 = (279.69668d0 + (36000.768930485d0 + 3.03d-4 * capt) * capt)
927  1 * convd
928 C
929 C Compute mean longitude of moon at beginning of day.
