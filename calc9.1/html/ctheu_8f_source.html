416 C THERY program structure
417 C--------------------------------------------------------------------------
418 C
419 C Call subroutine CONSEN to compute the delay and delay rate based on the
420 C Eubanks' Consensus relativity model. We put it in its own subroutine
421 C because it is a very long and detailed set of computations. 93JUL29, DG
422 C
423  Call consen ( datmc, earth, epbase, sitep, sitev,
424  1 sitea, sun, xmoon, star, tg2_tg1, dtg2_tg1,
425  2 delta_t_grav, d_delta_t_grav, delta_t_grav_sun,
426  3 d_delta_t_grav_sun, con_part, bend_par, sunplus )
427 C
428 C Add the axis offset corrections
429  tg2_tg1 = tg2_tg1 + daxoc(1,1) + daxoc(2,1)
430  dtg2_tg1 = dtg2_tg1 + daxoc(1,2) + daxoc(2,2)
431 C
432  consensus(1) = tg2_tg1 ! Consensus delay
433  consensus(2) = dtg2_tg1 ! Consensus rate
434 C
435 C Convert the theoretical delay to microseconds and split the double
436 C precision results into an integer microseconds portion and a
437 C submicroseconds portion.
438  xtg2_tg1 = tg2_tg1 * 1.d6
439  condel(1) = idint(xtg2_tg1 )
440  condel(2) = xtg2_tg1 - condel(1)
441  conrat = dtg2_tg1
442 C
443 C Put the Consensus model theoretical delays and rates into the database.
444  CALL put4 ('CONSNDEL ', condel, 2, 1, 1)
445  CALL put4 ('CONSNRAT ', conrat, 1, 1, 1)
446 C
447 C Compute and store the total gravitational light bending terms
448  con_cntrb(1) = delta_t_grav
449  con_cntrb(2) = d_delta_t_grav
450 C Compute and store the solar gravitational light bending terms
451  sun_cntrb(1) = delta_t_grav_sun
452  sun_cntrb(2) = d_delta_t_grav_sun
453 C
454 C Note: The Lcode 'CON CONT' will hold the TOTAL gravitational bending delay
455 C and rate from the Sun, Moon, Earth, and planets. 'SUN CONT' will hold
456 C the gravitational bending delay from the Sun only.
457 C 'CONSPART' will hold the TOTAL Consensus delay and rate partials with
458 C respect to Gamma.
459 C
460 C Put into the data base
461  CALL put4 ('CON CONT ', con_cntrb, 2, 1, 1)
462  CALL put4 ('SUN CONT ', sun_cntrb, 2, 1, 1)
463  CALL put4 ('CONSPART ', con_part, 2, 1, 1)
464  CALL put4 ('BENDPART ', bend_par, 2, 1, 1)
465 C The following is the higher order solar bending from the 1999 Conventions.
466 C It is already included in the total theoreticals ('CONSNDEL' and
467 C 'CONSNRAT') and the total solar bending contribution ('SUN CONT').
468  CALL put4 ('SUN2CONT ', sunplus, 2, 1, 1)
469 C
470 C--------------------------------------------------------------------------

473  WRITE (6, 9100 )

484 C
