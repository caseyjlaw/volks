159  SUBROUTINE ptdg ( SITLAT, SITLON, SITRAD, WOBXR, WOBYR, DIURNV,

162 C
163 C 4. PTDG
164 C
165 C 4.1 PTDG PROGRAM SPECIFICATION
166 C
167 C 4.1.1 PTDG Is the Pole tide module geometry section. It calculates the
168 C surface pole tide crustal displacements due to the oscillation of
169 C the Earth's pole about a mean secular pole. The pole tide station
170 C offsets are due to the solid Earth's response to the change in the
171 C centrifugal force due to the change in the pole position. There may
172 C be additional loading effects due to ocean pole tide effects, but
173 C these are not well understood, and not modelled here.
174 C
175 C Beginning with Calc 9.0, mean secular values for X-pole and Y-pole
176 C are removed before computing the pole tide. Thus the long term effect
177 C of pole tide, averaged over many Chandler cycles, should be null.
178 C Previously, a small velocity, from the cumulative effect of the
179 C secular movement of the pole, was imparted to each station.
180 C (maximums of ~.1 mm/year Up, and ~.03 mm/year horizontally.) The
181 C secular mean values of X-pole and Y-pole are computed as simple
182 C linear functions of time, using formulae provided by Harald Schuh.
183 C These equations are preliminary and may be changed slightly later,
184 C with the 2000 IERS Conventions, perhaps. Later adjustments can be
185 C mapped in using the partials w.r.t. X and Y. The X-mean and Y-mean
186 C values subtractred out are:
187 C X-mean(arc-sec) = +0.0039 + 0.00062*T
188 C Y-mean(arc-sec) = +0.1485 + 0.00333*T
189 C where T = number of years since 1945.85. The value of Y-mean used
190 C below will be the negative of the above, since the sign of Y-pole
191 C is flipped in the wobble module. 98.12.17
192 C
193 C At the request of Leonid Petrov, an Lcode contribution (PTOLDCON)
194 C to undo all the above has been added. It will be computed and
195 C ADD'ed in PTDC. 99.01.19
196 C
197 C 4.1.2 RESTRICTIONS - MUST BE CALLED AFTER SITG, AND DIRNL TO ENSURE
198 C THAT ALL OF THE VARIABLES THAT PTDG USES HAVE THE
199 C CORRECT VALUE.
200 C
201 C 4.1.3 REFERENCES - IERS Technical Note 21, IERS Conventions (1996),
202 C page 67, equation 22.
203 C E-mail from Harald Schuh, 17 Dec. 1998.
204 C
205 C 4.2 PTDG PROGRAM INTERFACE
206 C
207 C 4.2.1 CALLING SEQUENCE -
208 C INPUT VARIABLES:
209 C 1. SITLAT(2) - GEODETIC LATITUDE OF EACH SITE (RAD)
210 C 2. SITLON(2) - EAST LONGTIUDE OF EACH SITE (RAD)
211 C 3. SITRAD(2) - SPHERICAL EARTH RADIUS OF EACH SITE (M)
212 C 4. WOBXR - LONG PERIOD WOBBLE OFFSET ALONG GREENWICH
213 C MERIDIAN (RAD)
214 C 5. WOBYR - LONG PERIOD WOBBLE OFFSET ALONG LONGITUDE
215 C 90 DEG EAST (RAD)
216 C 6. DIURNV - DIURNAL SPIN RATE OF THE EARTH (RAD/SEC)
217 C 7. TCTOCF(3,3,2) - THE ROTATION MATRIX WHICH ROTATES THE
218 C TOPOCENTRIC REFERENCE SYSTEM TO THE CRUST
219 C FIXED REFERENCE SYSTEM AT EACH SITE
220 C 8. R2000(3,3,3) - THE COMPLETE CRUST FIXED TO J2000 ROTATION
221 C MATRIX AND ITS FIRST TWO CT TIME DERIVATIVES
222 C (UNITLESS, 1/SEC, 1/SEC**2)
223 C 9. CENT - Number of Julian centuries elapsed since the
224 C epoch January 1.5, 2000. (centuries)
225 C OUTPUT VARIABLES:
226 C 1. POLTDP(3,2) - THE CORRECTIONS TO THE J2000.0 GEOCENTRIC
227 C SITE POSITION VECTORS DUE TO POLE TIDAL
228 C EFFECTS AT EACH SITE. (M) [IN SOME CASES THIS
229 C IS ZEROED OUT DEPENDING ON THE VALUE OF THE
230 C POLE TIDE CONTROL FLAG KPTDC.]
231 C 2. POLTDV(3,2) - THE CORRECTIONS TO THE J2000.0 GEOCENTRIC
232 C SITE VELOCITY VECTORS DUE TO POLE TIDAL
233 C EFFECTS AT EACH SITE. (M/SEC) [IN SOME CASES
234 C THIS IS ZEROED OUT DEPENDING ON THE VALUE OF
235 C THE POLE TIDE CONTROL FLAG KPTDC.]
236 C
237 C 4.2.2 COMMON BLOCKS USED -
238 C
239  Real*8 ZPLTDP(3,2), ZPLTDV(3,2), ZPLDPX(3,2), ZPLDVX(3,2),
240  * zpldpy(3,2), zpldvy(3,2), x_mean, y_mean, dptdp(2,2)
241  COMMON / ptdcm / zpltdp, zpltdv, zpldpx, zpldvx, zpldpy, zpldvy,
242  * x_mean, y_mean, dptdp
243 C VARIABLES FROM:
244 C 1. ZPLTDP(3,2) - THE CORRECTIONS TO THE J2000.0 GEOCENTRIC SITE
245 C POSITION VECTORS DUE TO POLE TIDAL EFFECTS AT
246 C EACH SITE. (M)
247 C 2. ZPLTDV(3,2) - THE CORRECTIONS TO THE J2000.0 GEOCENTRIC SITE
248 C VELOCITY VECTORS DUE TO POLE TIDAL EFFECTS AT
249 C EACH SITE. (M/SEC)
250 C 3. ZPLDPX(3,2) - The partial derivative of the ZPLTDP with
251 C respective to the X-wobble offset. (m/arcsec)
252 C 4. ZPLDPY(3,2) - The partial derivative of the ZPLTDP with
253 C respective to the Y-wobble offset. (m/arcsec)
254 C 5. ZPLDVX(3,2) - The partial derivative of the ZPLTDV with
255 C respective to the X-wobble offset.
256 C (m/sec/arcsec)
257 C 6. ZPLDVY(3,2) - The partial derivative of the ZPLTDV with
258 C respective to the Y-wobble offset.
259 C (m/sec/arcsec)
260 C 7. X_mean, Y_mean - The current mean offsets of the pole, using
261 C Harald Schuh's formula.
262 C
263  Real*8 PI, TWOPI, HALFPI, CONVD, CONVDS, CONVHS, SECDAY
264  COMMON / cmath / pi, twopi, halfpi, convd, convds, convhs, secday
265  EXTERNAL cmathb
266 C VARIABLES 'FROM':
267 C 1. HALFPI - PI/2.D0 (UNITLESS)
268 C
269  include 'ccon.i'
270 C VARIABLES 'FROM':
271 C 1. KPTDC - THE POLE TIDE MODULE FLOW CONTROL FLAG.
272 C 2. KPTDD - THE POLE TIDE MODULE DEBUG OUTPUT FLAG.
273 C
274  include 'cobsn.i'
275 C Variables from:
276 C 1. Nzero - Set to 1 or 2 if station 1 or 2 is at the geocenter,
277 C then used downstream. Otherwise equals zero. For
278 C correlator usage.
279 C
280 C 4.2.3 PROGRAM SPECIFICATIONS -
281  Real*8 TCDISP(3,2), CFDISP(3,2), COLAT(2), SITLAT(2), SITLON(2),
282  . sitrad(2), tctocf(3,3,2), r2000(3,3,3), poltdp(3,2),
283  . poltdv(3,2), wobxr, wobyr, diurnv, cent,
284  . tcdspx(3,2), tcdspy(3,2), tyr, wobxd, wobyd
285  Integer*4 L, I
286 C
287 C 4.2.6 SUBROUTINE INTERFACE -
288 C CALLER SUBROUTINES: DRIVG
289 C CALLED SUBROUTINES: DCOS, DSIN, VECRT
290 C
291 C 4.2.7 CONSTANTS USED - XLOVEH, XLOVEL, DIURNV, WOBXR, WOBYR
292 C
293 C 4.2.8 PROGRAM VARIABLES -
294 C 1. TCDISP(3,2) - THE TOPOCENTRIC EARTH CRUSTAL DISPLACEMENTS AT EACH
295 C SITE (M)
296 C 2. CFDISP(3,2) - THE CRUST FIXED GEOCENTRIC EARTH CRUSTAL DISPLACEMENTS
297 C AT EACH SITE. (M)
298 C 3. COLAT(2) - COLATITUDE AT EACH SITE (USED FOR CONSISTENCY WITH REF
299 C 1) (RAD)
300 C
301 C 4.2.9 PROGRAMMER - TOM HERRING 07/01/84
302 C DAVID GORDON 01/03/85 (ADDED FLAG 2 CODING)
303 C Jim Ryan 89.07.08 Documentation simplified.
304 C David Gordon 11/24/93 Local site gravity computation
305 C copied from subroutine ETDG
306 C David Gordon 94.04.15 Changed to Implicit None
307 C David Gordon 94.04.27 Common block ETDCM changed (from
308 C Earth tide module).
309 C David Gordon 98.06.24 Changed common block ETDCM to match
310 C Earth tide module.
311 C David Gordon 98.08.04 Changed computations to match
312 C 1996 IERS Conventions; ETDCM not needed.
313 C Put in mods for Geocenter site (pole tide
314 C zeroed out).
315 C David Gordon 98.11.24 Added computations of the partial
316 C derivatives of the pole tide displacement
317 C w.r.t. X-pole and Y-pole. Partials added to
318 C common block PTDCM.
319 C David Gordon 98.12.17 Added variable CENT and computations
320 C for a mean pole X and Y offsets and their
321 C removal from the pole tide correction.
322 C David Gordon 99.01.19 Added X_mean, Y_mean, and DPTDP to
323 C Common /PTDCM/.
324 C
325 C PTDG PROGRAM STRUCTURE
326 C
327 C Compute the secular mean values of the X and Y pole positions
328 C Year and fraction thereof from 1945.85
329  tyr = cent*100.d0 + 2000.001368d0 - 1945.85d0
330 C Mean secular offsets (arc-seconds)
331  x_mean = ( 0.0039 + 0.00062*tyr)
332  y_mean = (-0.1485 - 0.00333*tyr)
333 C
334 C De-secularized X and Y (radians)
335  wobxd = wobxr - x_mean*convds
336  wobyd = wobyr - y_mean*convds
337 C
338 C
339 C The pole tide geometry for sites 1 and 2 are calculated
340 C separately by running through a loop twice.
341 C
342  DO 1200 l = 1,2
343 C
344 C Check for Geocenter site:
345  IF (l .eq. nzero) Then
346  Do i=1,3
347  tcdisp(i,l) = 0.d0
348  zpltdp(i,l) = 0.d0
349  zpltdv(i,l) = 0.d0
350  zpldpx(i,l) = 0.d0
351  zpldvx(i,l) = 0.d0
352  zpldpy(i,l) = 0.d0
353  zpldvy(i,l) = 0.d0
354  Enddo
355  Go to 1200
356  ENDIF
357 C
358 C Compute the colatitude of the site
359  colat(l) = halfpi - sitlat(l)
360 C
361 C Compute the IERS Conventions (1996) topocentric displacements.
362 C Up
363 C TCDISP(1,L) = -32.D0 * DSIN( 2.D0*COLAT(L) ) * 1.D-3 *
364 C * ( WOBXR*DCOS(SITLON(L)) + WOBYR*DSIN(SITLON(L)) )
365 C * / CONVDS
366 C East
367 C TCDISP(2,L) = 9.D0 * DCOS( COLAT(L) ) * 1.D-3 *
368 C * ( WOBXR*DSIN(SITLON(L)) - WOBYR*DCOS(SITLON(L)) )
369 C * / CONVDS
370 C North
371 C TCDISP(3,L) = 9.D0 * DCOS( 2.D0*COLAT(L) ) * 1.D-3 *
372 C * ( WOBXR*DCOS(SITLON(L)) + WOBYR*DSIN(SITLON(L)) )
373 C * / CONVDS
374 C
375 C
376 C Compute the IERS Conventions (1996) topocentric displacements with
377 C mean offsets removed.
378 C Up
379  tcdisp(1,l) = -32.d0 * dsin( 2.d0*colat(l) ) * 1.d-3 *
380  * ( wobxd*dcos(sitlon(l)) + wobyd*dsin(sitlon(l)) )
381  * / convds
382 C East
383  tcdisp(2,l) = 9.d0 * dcos( colat(l) ) * 1.d-3 *
384  * ( wobxd*dsin(sitlon(l)) - wobyd*dcos(sitlon(l)) )
385  * / convds
386 C North
387  tcdisp(3,l) = 9.d0 * dcos( 2.d0*colat(l) ) * 1.d-3 *
388  * ( wobxd*dcos(sitlon(l)) + wobyd*dsin(sitlon(l)) )
389  * / convds
390 C
391 C Rotate the displacements to the crust fixed geocentric system.
392  CALL vecrt ( tctocf(1,1,l), tcdisp(1,l), cfdisp(1,l) )
393 C Rotate the crust fixed geocentric displacements to the J2000.0 system.
394  CALL vecrt ( r2000(1,1,1), cfdisp(1,l), zpltdp(1,l) )
395 C Compute the contribution to the J2000.0 velocities due to the
396 C effect of the rotation of the earth.
397  CALL vecrt ( r2000(1,1,2), cfdisp(1,l), zpltdv(1,l) )
398 C
399 C Compute the partial derivatives of the displacements w.r.t. X and Y
400 C Recall that the sign of Y-pole was reversed a while back. Units will
401 C be meters/arc-sec.
402 C Up
403  tcdspx(1,l) = -32.d0 * dsin(2.d0*colat(l)) * 1.d-3 *
404  * dcos(sitlon(l))
405  tcdspy(1,l) = -32.d0 * dsin(2.d0*colat(l)) * 1.d-3 *
406  * dsin(sitlon(l))
407 C East
408  tcdspx(2,l) = 9.d0 * dcos(colat(l)) * 1.d-3 * dsin(sitlon(l))
409  tcdspy(2,l) = 9.d0 * dcos(colat(l)) * 1.d-3 * -dcos(sitlon(l))
410 C
411 C North
412  tcdspx(3,l) = 9.d0 * dcos( 2.d0*colat(l) ) * 1.d-3 *
413  * dcos(sitlon(l))
414  tcdspy(3,l) = 9.d0 * dcos( 2.d0*colat(l) ) * 1.d-3 *
415  * dsin(sitlon(l))
416 C
417 C Rotate to J2000 position and velocity offsets
418  CALL vecrt ( tctocf(1,1,l),tcdspx(1,l), cfdisp(1,l) )
419  CALL vecrt ( r2000(1,1,1), cfdisp(1,l), zpldpx(1,l) )
420  CALL vecrt ( r2000(1,1,2), cfdisp(1,l), zpldvx(1,l) )
421 C
422  CALL vecrt ( tctocf(1,1,l),tcdspy(1,l), cfdisp(1,l) )
423  CALL vecrt ( r2000(1,1,1), cfdisp(1,l), zpldpy(1,l) )
424  CALL vecrt ( r2000(1,1,2), cfdisp(1,l), zpldvy(1,l) )
425 C
426 C Close the loop over the sites.
427 C
428  1200 CONTINUE
429 C
430 C Check KPTDC to determine if the pole tide module is to be turned off
431 C and the contribution zeroed out.
432  IF ( kptdc .NE. 1 ) GO TO 300
433  DO 220 l = 1,2
434  DO 210 i = 1,3
435  poltdp(i,l) = 0.d0
436  poltdv(i,l) = 0.d0
437  zpltdp(i,l) = 0.d0
438  zpltdv(i,l) = 0.d0
439  zpldpx(i,l) = 0.d0
440  zpldvx(i,l) = 0.d0
441  zpldpy(i,l) = 0.d0
442  zpldvy(i,l) = 0.d0
443  210 CONTINUE
444  220 CONTINUE
445 C
446  300 CONTINUE
447 C
448 C Handle the normal case where the pole tide effect is to be applied
449 C to the theoretical and is stored as a contribution.
450  IF ( kptdc .NE. 0 ) GO TO 400
451  DO 320 l = 1,2
452  DO 310 i = 1,3
453  poltdp(i,l) = zpltdp(i,l)
454  poltdv(i,l) = zpltdv(i,l)
455  310 CONTINUE
456  320 CONTINUE
457 C
458  400 CONTINUE
459 C
460 C Handle the case where the pole tide is stored as a contribution
461 C but not applied to the theoretical.
462  IF ( kptdc .NE. 2 ) GO TO 500
463  DO 420 l = 1,2
464  DO 410 i = 1,3
465  poltdp(i,l) = 0.d0
466  poltdv(i,l) = 0.d0
467  410 CONTINUE
468  420 CONTINUE
469 C
470 C Check KPTDD for debug output.
471  500 IF ( kptdd .EQ. 0 ) GO TO 600
472 C
473  WRITE ( 6, 9100 )
474  9100 FORMAT ( /,1x, "Debug output for subroutine PTDG." )
475 C
476  WRITE ( 6, 9200 ) tcdisp, cfdisp, poltdp, poltdv, sitlat,
477  1 sitlon, wobxr, wobyr, r2000, tctocf
478 C
479  9200 FORMAT (1x, "TCDISP = ", 2 ( 3 ( d30.16, 10x ), /, 1x ),
480  1 "CFDISP = ", 2 ( 3 ( d30.16, 10x ), /, 1x ),
481  2 "POLTDP = ", 2 ( 3 ( d30.16, 10x ), /, 1x ),
482  3 "POLTDV = ", 2 ( 3 ( d30.16, 10x ), /, 1x ),
483  4 "SITLAT = ", 2 ( d30.16, 10x ), /, 1x,
484  5 "SITLON = ", 2 ( d30.16, 10x ), /, 1x,
