51  SUBROUTINE dirnl ( DATDCT, DPSI, DUT1AT, EPS, DEPS, FA, UT1,

55 C 1. DIRNL
56 C
57 C 1.1 DIRNL PROGRAM SPECIFICATION
58 C
59 C 1.1.1 DIRNL is the utility routine which computes the diurnal spin portion of
60 C the complete crust fixed to J2000.0 rotation matrix and its first two CT
61 C time derivatives. DIRNL also computes the diurnal angular velocity of
62 C the Earth, the Greenwich Mean Sidereal Time, and the Greenwich Apparent
63 C Siderial Time (GAST) and its CT time derivative.
64 C
65 C 1.1.2 RESTRICTIONS - NONE
66 C
67 C 1.1.3 REFERENCES - 1) 'THE EXPLANATORY SUPPLEMENT TO THE AMERICAN
68 C EPHEMERIS AND NAUTICAL ALMANAC", P.72-76,
69 C 2) AOKI, S. ET AL., "THE NEW DEFINITION OF UNIVERSAL
70 C TIME", ASTRON. ASTROPHYS., ????,1980.
71 C 3) McCarthy, D., IERS Technical Note 13, Paris 1992
72 C
73 C 1.2 DIRNL PROGRAM INTERFACE
74 C
75 C 1.2.1 CALLING SEQUENCE -
76 C
77 C INPUT VARIABLES:
78 C 1. DATDCT - THE PARTIAL DERIVATIVE OF ATOMIC TIME WITH
79 C RESPECT TO COORDINATE TIME. (SEC/SEC)
80 C 2. DPSI(2) - THE NUTATION IN LONGITUDE AND ITS CT
81 C TIME DERIVATIVE. (RAD, RAD/SEC)
82 C 3. DUT1AT - THE PARTIAL DERIVATIVE OF UT1 TIME WITH
83 C RESPECT TO ATOMIC TIME. (SEC/SEC)
84 C 4. EPS(2) - THE TRUE OBLIQUITY OF THE ECLIPTIC AND ITS
85 C CT TIME DERIVATIVE. (RAD, RAD/SEC)
86 C 5. DEPS(2) - THE NUTATION IN OBLIQUITY AND ITS CT TIME
87 C DERIVATIVE COMPUTED FROM WAHR OR TAKEN FROM
88 C THE DATA BASE. (RAD, RAD/SEC)
89 C 6. FA(5) - The fundamental nutation arguments (see NUTFA)
90 C 6. OMEGA - The mean longitude of the ascending node of the
91 C Moon. (arcsec!)
92 C 7. UT1 - THE UT1 TIME OF THE DAY. (SEC)
93 C 8. XJD - THE JULIAN DATE AT ZERO HOURS UTC OF THE
94 C DATE IN QUESTION. (DAYS)
95 C
96 C OUTPUT VARIABLES:
97 C 1. DIURNV - THE DIURNAL ANGULAR VELOCITY OF THE EARTH.
98 C (RAD/SEC)
99 C 2. GAST(2) - THE GREENWICH APPARENT SIDEREAL TIME AND
100 C ITS CT TIME DERIVATIVE. (RAD, RAD/SEC)
101 C 3. GMST - THE GREENWICH MEAN SIDEREAL TIME. (RAD)
102 C 4. GASTD - The difference in GAST (new-old version)
103 C 5. RS(3,3,3) - THE DIURNAL SPIN PORTION OF THE COMPLETE
104 C CRUST FIXED TO 2000.0 ROTATION MATRIX
105 C AND THE FIRST TWO CT TIME DERIVATIVES OF
106 C THAT MATRIX. (UNITLESS, 1/SEC, 1/SEC**2)
107 C
108  Real*8 DATDCT, DPSI(2), DUT1AT, EPS(2), DEPS(2), FA(5), UT1,
109  . xjd, diurnv, gast(2), gmst, gastd, rs(3,3,3)
110 C
111 C 1.2.2 COMMON BLOCKS USED -
112 C
113  Real*8 PI, TWOPI, HALFPI, CONVD, CONVDS, CONVHS, SECDAY
114  COMMON / cmath / pi, twopi, halfpi, convd, convds, convhs, secday
115  EXTERNAL cmathb
116 C
117 C VARIABLES 'FROM':
118 C 1. CONVHS - THE CONVERSION FACTOR OF RADIANS PER TIME-SECOND.
119 C (RAD/TIME-SEC)
120 C
121  include 'ccon.i'
122 C VARIABLES 'FROM':
123 C 1. KDIUC - THE DIURNAL SPIN UTILITY ROUTINE FLOW CONTROL FLAG.
124 C 2. KDIUD - THE DIURNAL SPIN UTILITY ROUTINE DEBUG OUTPUT FLAG.
125 C 3. KNUTC - THE NUTATION MODULE FLOW CONTROL FLAG. USED TO
126 C DETERMINE WHETHER OR NOT DPSI SHOULD BE ZEROED.
127 C
128  Real*8 GMSTC(4), SIDVEL(3), DPSIS(2), epsmd, omega, CENTJ, DJ2000,
129  . gmstmn, daysj, cent
130  Integer*4 I
131 C
132  DATA centj / 36525.d0 /,
133  1 dj2000 / 2451545.d0 /,
134  2 gmstc / + 24110.54841d0, + 8640184.812866d0,
135  3 + 0.093104d0, - 6.2d-6 /,
136  4 sidvel / + 1.002737909350795d0, + 5.9006d-11, - 5.9d-15 /

145 C CALLER SUBROUTINES: DRIVR

151 C 1. CENTJ - THE NUMBER OF COORDINATE TIME DAYS PER JULIAN
152 C CENTURY (CENTJ = 36525.D0 DAYS/CENTURY)
153 C 2. DJ2000 - THE JULIAN DATE OF THE EPOCH JANUARY 1.5, 2000.
154 C (DJ2000 = 2451545.D0 DAYS)
155 C 4. GMSTC(4) - THE CONSTANTS APPEARING IN TERMS 1-4 IN THE
156 C CALCULATION OF THE GREENWICH MEAN SIDEREAL TIME AT
157 C ZERO HOURS UT1 OF THE DATE IN QUESTION. (TIME-SEC)
158 C (SEE REFERENCES)
159 C ( GMSTC(1) = + 24110.54841D0,
160 C GMSTC(2) = + 8640184.812866D0,
161 C GMSTC(3) = + 0.093104D0,
162 C GMSTC(4) = - 6.2D-6 )
163 C 4. SIDVEL(3) - THE CONSTANTS APPEARING IN TERMS 1-3 IN THE
164 C CALCULATION OF THE EARTH'S DIURNAL ANGULAR VELOCITY.
165 C (TIME SEC/SEC) (SEE REFERENCES)
166 C ( SIDVEL(1) = + 1.002737909350795D0,
167 C SIDVEL(2) = + 5.9006D-11,
168 C SIDVEL(3) = - 5.9D-15 )
169 C
170 C 1.2.8 PROGRAM VARIABLES -
171 C 1. CENT - THE NUMBER OF JULIAN CENTURIES ELAPSED SINCE THE
172 C EPOCH JANUARY 1.5, 2000. (CENTURIES)
173 C 2. DAYSJ - THE NUMBER OF JULIAN DAYS ELAPSED SINCE THE EPOCH
174 C JANUARY 1.5, 2000. (DAYS)
175 C 3. GMSTMN - THE GREENWICH MEAN SIDEREAL TIME AT ZERO HOURS UT1
176 C OF THE DATE IN QUESTION. (RAD)
177 C 4. DPSIS(2) - THE VALUE OF NUTATION USED IN THE CALCULATION OF
178 C SIDEREAL TIME.
179 C 5. EPSMD - Mean obliquity of the ecliptic at date. (rad)
180 C 6. OMEGA - Mean longitude of the ascending node of the Moon
181 C (arcsec)
182 C
183 C 1.2.9 PROGRAMMER
184 C 770207 DALE MARKHAM
185 C 770718 PETER DENATALE
186 C 771128 BRUCE SCHUPLER
187 C 790201 BRUCE SCHUPLER
188 C 810804 CHOPO MA
189 C 870604 SAVITA GOEL
190 C 890726 Jim Ryan: Documentation simplified.
191 C 930901 Norbert Zacharias: Equation of equinoxes GAST update
192 C
193 C DIRNL Program Structure
194 C
195 C Compute mean obliquity of date, req. for EQE update in DIRNL
196  epsmd = eps(1) - deps(1)
197 C Mean longitude of ascending node of the Moon
198  omega = fa(5)
199 
200 C Compute the number of Julian days elapsed since the epoch
201 C JANUARY 1.5, 2000 to 0 hours UT1 of the date in question.
202  daysj = xjd - dj2000
203 C
204 C Compute the number of Julian centuries elapsed since Epoch January 1.5, 2000
205  cent = daysj / centj
206 C
207 C Compute the diurnal angular velocity of the Earth.
208  diurnv = ( sidvel(1) + sidvel(2) * cent + sidvel(3) * cent**2 )
209  1 * convhs
210 C
211 C Compute the Greenwich Mean Sidereal Time at 0 hours UT1 of the date
212 C in question. (radians)
213  gmstmn = ( gmstc(1) + gmstc(2) * cent + gmstc(3) * cent**2 +
214  . gmstc(4) * cent**3 ) * convhs
215 C
216 C Compute the Greenwich Mean Sidereal Time.
217  gmst = gmstmn + diurnv * ut1
218  gmst = dmod(gmst,twopi)
219 C
220 C Check to see if the nutation in longitude is to be turned off.
221  DO i=1,2
222  IF (knutc .EQ. 1) dpsis(i) = 0.0d0
223  IF (knutc .NE. 1) dpsis(i) = dpsi(i)
224  ENDDO
225 C
226 C Compute the Greenwich Apparent Sidereal Time by calling the geometry part of
227 C the equation of equinox procedure.
228  CALL eqeg (gmst, dpsis(1), eps(1), epsmd, omega,
229  . gast, gastd)
230 C
231 C Compute the time derivative of Greenwich Apparent Siderial Time
232  gast(2) = diurnv * dut1at * datdct
233  1 + dcos( eps(1) ) * dpsis(2)
234  2 - dsin( eps(1) ) * dpsis(1) * eps(2)
235 C
236 C Construct the diurnal spin matrix and its first two CT time derivatives.
237 C
238 C Construct the diurnal spin matrix.
239  CALL rotat ( -gast(1), 3, rs(1,1,1) )
240 C
241 C Construct the first CT time derivative of the diurnal spin matrix
242  CALL drott ( -gast(1), -gast(2), 3, rs(1,1,2) )
243 C
244 C Construct the second CT time derivative of the diurnal spin matrix.
245  CALL ddrot ( -gast(1), gast(2)**2, 3, rs(1,1,3) )
246 C
247 C****************************************************************************
248 C Check KDIUD for debug output.
249  IF ( kdiud .NE. 0 ) THEN
250  WRITE ( 6, 9)
251  9 FORMAT (1x, "Debug output for utillity DIRNL." )
252  WRITE(6,8)' CENT ',cent
253  8 FORMAT(a,4d25.16/(7x,5d25.16))
254  WRITE(6,8)' CENTJ ',centj
255  WRITE(6,8)' DJ2000 ',dj2000
256  WRITE(6,8)' DPSIS ',dpsis
257  WRITE(6,8)' CONVHS ',convhs
258  WRITE(6,8)' DAYSJ ',daysj
259  WRITE(6,8)' GMSTC ',gmstc
260  WRITE(6,8)' GMSTMN ',gmstmn
261  WRITE(6,8)' SIDVEL ',sidvel
262 C
263  WRITE ( 6, 9200 ) datdct, dpsi, dut1at, eps, ut1, xjd,
264  1 diurnv, gast, gmst, rs
265  9200 FORMAT (1x, "DATDCT = ", d30.16, /, 1x,
266  1 "DPSI = ", 2 ( d30.16, 10x ), /, 1x,
267  2 "DUT1AT = ", d30.16, /, 1x,
268  3 "EPS = ", 2 ( d30.16, 10x ), /, 1x,
269  4 "UT1 = ", d30.16, /, 1x,
270  5 "XJD = ", d30.16, /, 1x,
271  6 "DIURNV = ", d30.16, /, 1x,
272  7 "GAST = ", 2 ( d30.16, 10x ), /, 1x,
273  8 "GMST = ", d30.16, /, 1x,
274  9 "RS = ", 9 ( 3 ( d30.16, 10x ), /, 1x ) )
275 C
276  ENDIF ! debug output flag
277 C
278  Return
279  END

282  SUBROUTINE eqeg (gmst,dpsi,eps,epsmd,omega, gast,gastd)
283  IMPLICIT NONE
284 C
285 C This subroutine computes the geometry section of the equation of equinox
286 C procedure.
287 C
288 C References: D.McCarthy, IERS Technical Note 13, Paris 1992
289 C
290 C Calling sequence:
291 C input variables:
292 C gmst = Greenwich mean siderial time (radian)
293 C dpsi = nutation in longitude (radian)
294 C eps = true obliquity of the ecliptic (radian)
295 C epsmd = obliquity of the mean ecliptic of date (radian)
296 C omega = mean longitude of the asc. node of the moon (arcsec)
297 C
298 C output variables:
299 C GAST(1)=Greenwich apparent siderial time (New or Old depending on KDIUC)
300 C (radians)
301 C GASTD = Difference in GAST of two versions (New-Old) (radians)
302 C
303  REAL*8 gmst, dpsi, eps, epsmd, omega, gast(2), gastd
304  REAL*8 old_gast, aswas_gast, new_gast,
305  . eqe_const(2), omega_rad
306  REAL*8 pi, twopi, halfpi, convd, convds, convhs, secday
307  COMMON /cmath/ pi, twopi, halfpi, convd, convds, convhs, secday
308  EXTERNAL cmathb
309 C
310 C Variables from: convds = convert arcsec to radian
311 C
312  include 'ccon.i'
313 C Variables from:
314 C KDIUC - Flow control flag for EQE routines
315 C = 0, GAST defined by new equation of equinoxes definition
316 C = 1, GAST defined by old equation of equinoxes definition
317 C KDIUD - Debug flag for EQE routines
318 C
319 C Data base access: None
320 C
321 C Subroutine interface:
322 C caller subr.: DIRNL
323 C called subr.: None
324 C
325 C Program variables:
326 C eqe_const = Constants in new version of GAST formula
327 C depending on sine of omega (arcsec)
328 C aswas_gast = GAST as calculated incorrectly before (CALC 7.6 and
329 C earlier). (radian)
330 C old_gast = GAST as calculated before (CALC 7.6 and earlier) but
331 C corrected to use mean obliquity of date, to be used until
332 C 1997.02.26. (radians)
333 C omega_rad = Omega (longit. of asc. node of moon) (radians)
334 C new_gast = GAST for use after 1997.02.26 as recommended by the IERS.
335 C
336 C Programmer:
337 C 930901 Norbert Zacharias: first implementation, equat.of eq. GAST
338 C update.
339 C 94.01.04 David Gordon - Clean up documentation, change sign of GAST
340 C correction.
341 C 98.01.20 David Gordon - Updated to use new definition EQE
342 C
343  DATA eqe_const / 0.00264d0, 0.000063d0 / ! in arc sec
344 C
345 C compute old GAST as before, use true obliquity eps as before
346  aswas_gast = gmst + dpsi * dcos(eps)
347 C
348 C compute old GAST (formula in effect until 19970226)
349 C but with correct epsilon (mean obliquity of date)
350  old_gast = gmst + dpsi * dcos(epsmd)
351 C
352 C compute new GAST as recomm. in IERS techn.note
353  omega_rad = omega * convds
354  new_gast = gmst + dpsi * dcos(epsmd)
355  . + eqe_const(1) * convds * dsin(omega_rad)
356  . + eqe_const(2) * convds * dsin(2.0d0*omega_rad)
357 C
358 C Calc 9, use new definition
359  If (kdiuc.eq.0) gast(1) = new_gast
360 C For diehards, allow using old definition
361  If (kdiuc.eq.1) gast(1) = old_gast
362 C Define difference, Gastd, as (New - Old)
363  gastd = new_gast - old_gast
364 C
365 C check debug output flag
366  IF (kdiud.NE.0) THEN
367  WRITE (6,'(1x,a)') 'debug output for EQEG'
368  WRITE (6,'(1x,a,d25.16)') 'gmst = ', gmst
369  WRITE (6,'(1x,a,d25.16)') 'dpsi = ', dpsi
370  WRITE (6,'(1x,a,d25.16)') 'eps = ', eps
371  WRITE (6,'(1x,a,d25.16)') 'epsmd = ', epsmd
372  WRITE (6,'(1x,a,d25.16)') 'omega = ', omega
373  WRITE (6,'(1x,a,d25.16)') 'aswas_gast = ', aswas_gast
374  WRITE (6,'(1x,a,d25.16)') 'old_gast = ', old_gast
375  WRITE (6,'(1x,a,d25.16)') 'new_gast = ', new_gast
376  WRITE (6,'(1x,a,d25.16)') 'gast(1) = ', gast(1)
377  WRITE (6,'(1x,a,d25.16)') 'aswas-old = ', aswas_gast-old_gast
378  WRITE (6,'(1x,a,d25.16)') 'gastd = ', gastd
379  ENDIF
380 C
381  Return
382  END

384  SUBROUTINE dirnc (gastd, dut1p)
385  IMPLICIT NONE
386 C
387 C This subr. computes the contributions to the delay and delay rate due to the
388 C different versions of the equation of equinox procedure. The New version
389 C (to be used after 26 Feb 1997) becomes the default in Calc 9.0. To get back
390 C to the old definition, simply add the contribution Lcode OLDEQCON during
391 C SOLVE analysis.
392 C
393 C Users can still use the old EQE definition by setting KDIUC=1. In that case,
394 C a diferent contribution goes into the database, 'NEWEQCON'. When NEWEQCON
395 C is added to the delays and rates in this case, they will be converted to
396 C the new definition. Don't count on this being supported in SOLVE though.
397 C
398 C References: D.McCarthy, IERS Technical Note 13, Paris 1992
399 C E-Mail Memo, 'UT1 Errors', D. Gordon, 10 April, 1997
400 C
401 C Calling sequence:
402 C Input Variables:
403 C GASTD = Differ. in GAST of the two versions (new-old) (radian)
404 C DUT1P(2,2) = Partial derivatives of delay and rate wrt (A1-UT1)
405 C (First index - delay and rate, second index - first and
406 C second derivatives) (s/s, s/s**2)
407 C [Note: The partial we want here is actually the delay and rate
408 C derivatives wrt GAST. We approximate this by dividing
409 C dut1p by -1.0027379 sidereal days/solar day.]
410 C
411  REAL*8 gastd, dut1p(2,2), deqec(2)
412  REAL*8 pi,twopi, halfpi, convd, convds, convhs, secday
413  COMMON /cmath/ pi,twopi,halfpi, convd, convds,
414  . convhs, secday
415  EXtERNAL cmathb
416 C Variables From:
417 C Convhs = conversion factor (radian/sec.of time)
418 C
419  include 'ccon.i'
420 C Variables From:
421 C KDIUC - Flow control flag for EQE routines
422 C = 0, GAST defined by new equation of equinoxes definition
423 C = 1, GAST defined by old equation of equinoxes definition
424 C KDIUD - debug flag for EQE routines
425 C
426 C Data base access:
427 C PUT4 GASTD = differ. in GAST (new - old) (radian)
428 C PUT4 DEQEC(2) = contrib. of GASTD to delay and delay rate
429 C (sec, sec/sec)
430 C Subroutine interface:
431 C Caller Subr.: DRIVR
432 C Called Subr.: None
433 C
434 C Programmer:
435 C 930901 Norbert Zacharias: first implementation
436 C 94.01.04 David Gordon - Fixed up documentation, sign of contributions
437 C flipped in EQEG.
438 C 94.06.02 David Gordon - Changed PUTR to PUT4.
439 C 95.02.09 David Gordon - Corrected PUT4's of 'EQE DIFF' and 'EQE CONT'.
440 C Only 8 charcters allocated, changed to 14. Bug
441 C found by Bill Petrachenko using a Sun workstation
442 C at the Dominion Radio Astrophysical Obs. (DRAO).
443 C 98.01.20 David Gordon - Corrected old (Calc 8.x) delay and rate
444 C contribution (Deqec) by dividing by -1.0027379
445 C siderial days/solar day. Changed database Lcode to
446 C 'NEWEQCON' (from 'EQE CONT') and made it the
447 C non-default. Added 'OLDEQCON' Lcode contribution
448 C and made it the default.
449 C
450 C Calculate contributions to delay and rate
451 C
452 C Default case, contribution to convert from New to Old EQE definition
453  If (kdiuc .eq. 0) then
454  deqec(1) = -gastd * dut1p(1,1) / (-1.0027379 * convhs)
455  deqec(2) = -gastd * dut1p(2,1) / (-1.0027379 * convhs)
456  CALL put4 ('OLDEQCON ', deqec, 2,1,1)
457  Endif
458 C
459 C Non-default case, contribution to convert from Old to New EQE definition
460  If (kdiuc .eq. 1) then
461 C Correct Calc 8.2 error by dividing by -1.0027379
462  deqec(1) = gastd * dut1p(1,1) / (-1.0027379 * convhs)
463  deqec(2) = gastd * dut1p(2,1) / (-1.0027379 * convhs)
464  CALL put4 ('NEWEQCON ', deqec, 2,1,1)
465  Endif
466 C
467 C Output EQE difference to data base
468  CALL put4 ('EQE DIFF ', gastd, 1,1,1)
469 C
470 C check flag for debug output:
471  IF (kdiud.NE.0) THEN
472  WRITE (6,'(1x,a)') 'debug output for EQEC'
473  WRITE (6,'(1x,a,2d25.16)')
474  . 'dut1p = ', dut1p(1,1), dut1p(2,1)
475  WRITE (6,'(1x,a,2d25.16)')
476  . 'gastd, convhs = ', gastd, convhs
477  WRITE (6,'(1x,a,2d25.16)') 'deqec = ', deqec
478  ENDIF
479 C
480  Return
481  END
