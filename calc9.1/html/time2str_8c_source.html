19 static int ltostr /* convert long to string */
20  (
21  long number, /* number to be converted */
22  char *pOutStr, /* receiving string */
23  int width, /* minimum width of numeric field,
24  positive width => use leading blank fill,
25  negative => leading zero fill */
26  int sign /* only the sign of this int is important,
27  negative => places '-' in output */
28  )
29 /*
30  * RETURNS number of characters that have been appended to string
31  *
32  * Output digits are always right-justified in the field. BE CAREFUL - If
33  * 'width' is less than the number of digits needed to express the number, the
34  * field will be expanded to the exact number of characters required.
35  *
36  * A minus sign is prefixed if either 'number' or 'sign' is negative. This
37  * permits a 'signed zero' to be generated. The minus sign is the first
38  * character in the string if leading zeros are requested, else it is the
39  * character immediately preceeding the most significant digit.
40  *
41  * The output string DOES NOT have a terminating null.
42  */
43 {
44  char *preverse;
45  char fillchar;
46  char tempch;
47  int nochar = 0;
48 
49  /* if input number is negative ... */
50  if (number < 0)
51  {
52  sign = -1;
53  number = -number;
54  }
55 
56  /* set fillchar = proper fill character */
57  if (width < 0)
58  {
59  width = -width;
60  fillchar = '0';
61  }
62  else
63  fillchar = ' ';
64 
65  /* if input number is negative with leading zeroes fill,
66  insert minus sign in first character of string */
67  if (sign < 0 && fillchar == '0')
68  {
69  *pOutStr++ = '-';
70  nochar++;
71  sign = 0;
72  }
73 
74  /* record pointer where string reversal will begin */
75  preverse = pOutStr;
76 
77  /* convert input number to ascii in reverse order */
78  do
79  {
80  *pOutStr++ = '0' + number % 10;
81  nochar++;
82  } while ((number /= 10) > 0);
83 
84  /* if input number is negative with leading blank fill,
85  insert minus sign in front of most significant digit */
86  if (sign < 0 && fillchar == ' ')
87  {
88  *pOutStr++ = '-';
89  nochar++;
90  }
91 
92  /* insert enough fill characters to fill minimum field width */
93  while (nochar < width)
94  {
95  *pOutStr++ = fillchar;
96  nochar++;
97  }
98 
99  /* point to last inserted character */
100  pOutStr--;
101 
102  /* digits were generated in reverse order, so re-reverse them */
103  while (preverse < pOutStr)
104  {
105  tempch = *preverse;
106  *preverse++ = *pOutStr;
107  *pOutStr-- = tempch;
108  }
109 
110  /* return number of characters inserted */
111  return (nochar);
112 }

117  double angle, /* input angle in radians */

120  BOOL roundFlag /* TRUE- output string is rounded, FALSE-truncated */

122 /*
123  * RETURNS original pointer to given string
124  *
125  * Converts a floating point number presumed to contain an angle in radians to
126  * a string according to instructions in the format string. The output string
127  * format can be either degrees (default form is iiidjj'kk.f") or time
128  * (default form is iihjjmkk.ffs) depending on the format string contents.
129  * The jj (minutes) and kk (integer seconds) fields are always two digits.
130  * The widths of the iii (degrees or hours) and ff (fractional seconds) fields
131  * are controlled by the format string.
132  *
133  * The format string is a null-terminated string of the general form "dm.n" or
134  * "hm.n" where m and n are digit strings. The forms "d", "h" (use default
135  * values for m and n), "dm", "hm" (use given m and default n), "d.n", and
136  * "h.n" (use default m and given n) are also permitted.
137  *
138  * The digit string m controls the width of the degrees/hours field in the
139  * output string. If m does not specify sufficient digits to represent the
140  * degrees/hours, the field will be widened to the exact number of digits
141  * needed. If the first digit of m is zero, zero fill will be used; otherwise
142  * blank fill will be used.
143  *
144  * The digit string n controls the number of decimal places in the fractional
145  * portion of the seconds field. If n is zero, no seconds fraction and no
146  * decimal point will appear in the seconds field.
147  *
148  * If a ':' is included after the 'h', colons are used as a delimiter between
149  * the hours & minutes and between the minutes and seconds, a NULL is placed
150  * after the last digit of the seconds. All other 'hour' formatting rules
151  * described above still apply.
152  *
153  * EXAMPLE FORMAT STRINGS AND THEIR RESULTS --
154  * "d", or indecipherable string -- 111d22'33.3"
155  * "h" -- 11h22m33.33s
156  * "d4" -- 1d22'33.3"
157  * "h.1" -- 11h22m33.3s
158  * "h:.1" -- 11:22:33.3
159  * "d03.2" -- 001d22'33.33"
160  */
161 {
162  char *poriginal = pOutStr; /* hold output string address for return */
163  char units_char[4]; /* field separation chars (d'" or hms) */
164  int m; /* degrees/hours field width */
165  int n; /* fractional seconds field width */
166  BOOL lzero_flag = FALSE; /* TRUE => use leading zeros fill */
167  int sign_flag = 1; /* 1=positive, -1=negative angle */
168  long long_tmp;
169  int i;
170 
171  /* decode measure type from format string */
172  if (*pFormat == 'h')
173  { /* hour format */
174  angle *= HOUR_RAD; /* convert angle to hours */
175  if (*(pFormat+1) == ':') /* if colon delimiter specified */
176  {
177  strcpy (units_char, "::\0"); /* set field separator chars */
178  pFormat++; /* account for extra char */
179  }
180  else /* use h m s format */
181  strcpy (units_char, "hms"); /* set field separator chars */
182  m = 2; /* default hours field width */
183  n = 2; /* default frac seconds field width */
184  }
185  else
186  { /* degrees (or default) format */
187  angle *= DEG_RAD; /* convert angle to degrees */
188  strcpy (units_char, "d'\""); /* set field separator chars */
189  m = 3; /* default degrees field width */
190  n = 1; /* default frac seconds field width */
191  }
192 
193  /* advance to next char if not at end */
194  if (*pFormat != '\0')
195  pFormat++;
196 
197  /* decode given degree/hour field width (if any) */
198  if (isdigit (*pFormat))
199  {
200  if (*pFormat == '0')
201  {
202  lzero_flag = TRUE; /* leading zeros fill */
203  pFormat++;
204  }
205  if (isdigit (*pFormat))
206  {
207  m = 0;
208  while (isdigit (*pFormat))
209  m = m * 10 + *pFormat++ - '0';
210  }
211  }
212 
213  /* decode given fractional seconds field width (if any) */
214  if (*pFormat == '.')
215  {
216  pFormat++; /* skip over . */
217  if (isdigit (*pFormat))
218  {
219  n = 0;
220  while (isdigit (*pFormat))
221  n = n * 10 + *pFormat++ - '0';
222  }
223  }
224 
225  /* if value is negative set sign flag and convert to positive */
226  if (angle < 0.0)
227  {
228  sign_flag = -1;
229  angle = -angle;
230  }
231 
232  /* add rounding fraction to angle appropriate for desired output */
233  if (roundFlag)
234  {
235  double round = 1.3888889E-4; /* 1/2 second = 1/7200 */
236  for (i = 0; i < n; i++)
237  round /= 10.0;
238  angle += round;
239  }
240 
241  /* calculate and place degree/hours in output string
242  here units of angle are either hours or degrees set according to format */
243  long_tmp = angle; /* note use of truncation */
244  if (lzero_flag) /* neg field width => use leading zeros */
245  m = -m;
246  pOutStr += ltostr (long_tmp, pOutStr, m, sign_flag);
247  *pOutStr++ = units_char[0];
248 
249  /* calculate and place minutes in output string */
250  angle = (angle - long_tmp) * 60.0;
251  long_tmp = angle;
252  pOutStr += ltostr (long_tmp, pOutStr, -2, 0);
253  *pOutStr++ = units_char[1];
254 
255  /* calculate and place integer seconds in output string */
256  angle = (angle - long_tmp) * 60.0;
257  long_tmp = angle;
258  pOutStr += ltostr (long_tmp, pOutStr, -2, 0);
259 
260  /* calculate and place fractional seconds in output string */
261  if (n != 0)
262  {
263  *pOutStr++ = '.';
264  angle -= long_tmp; /* angle = fractional part of secs */
265  for (i = 0; i < n; i++)
266  angle *= 10.0;
267  pOutStr += ltostr ((long)angle, pOutStr, -n, 0);
268  }
269  *pOutStr++ = units_char[2];
270  *pOutStr++ = '\0';
271 
272  return (poriginal);
273 }
