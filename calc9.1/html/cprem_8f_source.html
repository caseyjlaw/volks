291  Real*8 DRP1DT(3,3), DRP2DT(3,3), DRP3DT(3,3), DRPDT1(3,3),

312 C RP2. (1/SEC)

314 C RP3. (1/SEC)

316 C PRECESSION MATRIX. (1/SEC)
317 C 6. DRPDT2(3,3) - THE SECOND TERM OF THE CT TIME DERIVATIVE OF
318 C THE PRECESSION MATRIX. (1/SEC)
319 C 7. DRPDT3(3,3) - THE THIRD TERM OF THE CT TIME DERIVATIVE OF THE
320 C PRECESSION MATRIX. (1/SEC)
321 C 8. DYSJ - THE NUMBER OF JULIAN DAYS ELAPSED SINCE THE
322 C EPOCH J2000.0. (DAYS)
323 C 9. HC - THE TERM WHICH SHOWS THE DEPENDENCE OF SMALL
324 C CHANGES IN THE PRECESSION CONSTANT FROM ITS
325 C NOMINAL VALUE IN THE EPOCH J2000.0 ON THE
326 C PRECESSIONAL ELEMENTS ZEE AND ZETA. (RAD)
327 C 10. HS - THE TERM WHICH SHOWS THE DEPENDENCE OF SMALL
328 C CHANGES IN THE PRECESSION CONSTANT FROM ITS
329 C NOMINAL VALUE IN THE EPOCH J2000.0 ON THE
330 C PRECESSIONAL ELEMENT THETA. (RAD)

336 C ELEMENT ZETA. (RAD/SEC)
337 C
338 C 4.2.9 PROGRAMMER - DALE MARKHAM 01/13/77
339 C PETER DENATALE 07/12/77
340 C BRUCE SCHUPLER 11/10/77
341 C BRUCE SCHUPLER 06/06/78
342 C CHOPO MA 08/04/81
343 C Jim Ryan 89.07.07 Documentation simplified.
344 C David Gordon 94.04.14 Converted to Implicit None
345 C
346 C PREG PROGRAM STRUCTURE
347 C
348 C Calculate the relevant time variables.
349 C
350 C Calculate the number of Julian days elapsed since epoch J2000.0.
351  dysj = xjd + ct - dj2000
352 C
353 C Calculate the number of Julian centuries elapsed since epoch J2000.0.
354  centr = dysj / centj
355 C
356 C Calculate the CT time derivative of CENTR.
357  dcentr = 1.d0 / ( centj * secday )
358 C
359 C Calculate the precessional elements ZETA, ZEE, and THETA and
360 C their CT time derivatives.
361 C
362 C Compute the terms which show the dependence of small changes of the
363 C precession constant from its nominal value in the epoch J2000.0 on the
364 C precessional elements.
365  hc = 0.d0
366  hs = 0.d0
367  IF ( dabs( precon - hepoch ) .LT. 1.d-4 ) GO TO 220
368  hc =( precon - hepoch ) * dcos( epsmnr ) * convds *
369  1 centr / 2.d0
370  hs =( precon - hepoch ) * dsin( epsmnr ) * convds *
371  1 centr
372 C
373 C Compute ZETA and its CT time derivative.
374  220 zeta = ( czeta(1) * centr
375  1 + czeta(2) * centr**2
376  2 + czeta(3) * centr**3 ) * convds
377  3 + hc
378  tzeta = ( czeta(1)
379  1 + czeta(2) * ( 2.d0 * centr )
380  2 + czeta(3) * ( 3.d0 * centr**2 ) ) * convds * dcentr
381  3 + ( hc / centr ) * dcentr
382 C
383 C Compute ZEE and its CT time derivative.
384  zee = ( czee(1) * centr
385  1 + czee(2) * centr**2
386  2 + czee(3) * centr**3 ) * convds
387  3 + hc
388  tzee = ( czee(1)
389  1 + czee(2) * ( 2.d0 * centr )
390  2 + czee(3) * ( 3.d0 * centr**2 ) ) * convds * dcentr
391  3 + ( hc / centr ) * dcentr
392 C
393 C Compute THETA and its CT time derivative.
394  theta = ( ctheta(1) * centr
395  1 + ctheta(2) * centr**2
396  2 + ctheta(3) * centr**3 ) * convds
397  3 + hs
398  ttheta = ( ctheta(1)
399  1 + ctheta(2) * ( 2.d0 * centr )
400  2 + ctheta(3) * ( 3.d0 * centr**2 ) ) * convds * dcentr
401  3 + ( hs / centr ) * dcentr
402 C
403 C Construct the precession matrix.
404 C
405 C Construct the rotation matrix which rotates about the current north
406 C celestial pole by an angle equal to ZEE+HALFPI.
407  CALL rotat ( zee+halfpi, 3, rp1 )
408 C
409 C Construct the rotation matrix which rotates about the ascending node
410 C of the equator by an angle equal to TO -THETA.
411  CALL rotat ( -theta, 1, rp2 )
412 C
413 C Construct the rotation matrix which rotates about the J2000.0 north
414 C celestial pole by an angle equal to ZETA - HALFPI.
415  CALL rotat ( zeta-halfpi, 3, rp3 )
416 C
417 C Complete the construction of the precession matrix.
418  CALL mmul3 ( rp3, rp2, rp1, rp(1,1,1) )
419 C
420 C Construct the CT time derivative of the precession matrix.
421 C
422 C Construct the CT time derivative of the rotation matrix RP1.
423  CALL drott ( zee+halfpi, tzee, 3, drp1dt )
424 C
425 C Construct the CT time derivative of the rotation matrix RP2.
426  CALL drott ( -theta, -ttheta, 1, drp2dt )
427 C
428 C Construct the CT time derivative of the rotation mariix RP3.
429  CALL drott ( zeta-halfpi, tzeta, 3, drp3dt )
430 C
431 C Compute the three terms which make up the CT time derivative of the
432 C precession matrix.
433  CALL mmul3 ( drp3dt, rp2, rp1, drpdt1 )
434  CALL mmul3 ( rp3, drp2dt, rp1, drpdt2 )
435  CALL mmul3 ( rp3, rp2, drp1dt, drpdt3 )
436 C
437 C Complete the construction of the CT time derivative of the precession
438 C matrix.
439  CALL madd3 ( drpdt1, drpdt2, drpdt3, rp(1,1,2) )
440 C
441 C Check KPREC to determine if the precession module is to be turned off.
442  IF ( kprec .NE. 1 ) GO TO 600
443 C
444 C It's turned off, so set the position portion of the precession matrix to
445 C the identity matrix and zero out the velocity matrix.
446  CALL rotat ( 0.d0, 3, rp(1,1,1) )
447  DO 500 n=1,3
448  DO 500 m=1,3
449  rp(m,n,2) = 0.0d0
450  500 CONTINUE
451 C
452 C Check KPRED for debug.
453  600 IF ( kpred .EQ. 0 ) GO TO 700
454  WRITE ( 6, 9)
455  9 FORMAT (1x, "Debug output for subroutine PREG." )
456 C
457  WRITE(6,8)' CENTR ',centr
458  8 FORMAT(a,4d25.16/(7x,5d25.16))
459  WRITE(6,8)' CENTJ ',centj
460  WRITE(6,8)' CONVDS ',convds
461  WRITE(6,8)' CTHETA ',ctheta
462  WRITE(6,8)' CZEE ',czee
463  WRITE(6,8)' CZETA ',czeta
464  WRITE(6,8)' DCENTR ',dcentr
465  WRITE(6,8)' DJ2000 ',dj2000
466  WRITE(6,8)' DRP1DT ',drp1dt
467  WRITE(6,8)' DRP2DT ',drp2dt
468  WRITE(6,8)' DRP3DT ',drp3dt
469  WRITE(6,8)' DRPDT1 ',drpdt1
470  WRITE(6,8)' DRPDT2 ',drpdt2
471  WRITE(6,8)' DRPDT3 ',drpdt3
472  WRITE(6,8)' DYSJ ',dysj
473  WRITE(6,8)' HALFPI ',halfpi
474  WRITE(6,8)' HC ',hc
475  WRITE(6,8)' HEPOCH ',hepoch
476  WRITE(6,8)' HS ',hs
477  WRITE(6,8)' PRECON ',precon
478  WRITE(6,8)' RP1 ',rp1
479  WRITE(6,8)' RP2 ',rp2
480  WRITE(6,8)' RP3 ',rp3
481  WRITE(6,8)' SECDAY ',secday
482  WRITE(6,8)' THETA ',theta
483  WRITE(6,8)' TTHETA ',ttheta
484  WRITE(6,8)' TZEE ',tzee
485  WRITE(6,8)' TZETA ',tzeta
486  WRITE(6,8)' ZEE ',zee
487  WRITE(6,8)' ZETA ',zeta
488 C
489  WRITE ( 6, 9200 ) ct, epsmnr, xjd, rp
490  9200 FORMAT (1x, "CT = ", d30.16, /, 1x,
491  1 "EPSMNR= ", d30.16, /, 1x,
492  2 "XJD = ", d30.16, /, 1x,
493  3 "RP = ", 6 ( 3 ( d30.16, 10x ), /, 1x ) )
494 C
495 C 7. NORMAL PROGRAM CONCLUSION.

507 C
