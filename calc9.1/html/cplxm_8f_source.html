330 C 5.3 PLXP PROGRAM STRUCTURE
331 C

344  ddcdpi = sunvct(3)*cd-sunvct(1)*cra*sd-sunvct(2)*sra*sd
345 C
346 C Apply the chain rule.
347  dplxp(1)=dradpi*dstrp(1,1)+ddcdpi*dstrp(2,1)
348  dplxp(2)=dradpi*dstrp(1,2)+ddcdpi*dstrp(2,2)

354 C DRADPI = DRADPI * CONVDS
355 C DDCDPI = DDCDPI * CONVDS
356 C dK(1) = -SD*DDCDPI*CRA - CD*SRA*DRADPI
357 C dK(2) = -SD*DDCDPI*SRA + CD*CRA*DRADPI
358 C dK(3) = CD*DDCDPI
359 C print dK in arc-seconds
360 C WRITE(6,8)'test/dK: ', dK(1)/CONVDS, dK(2)/CONVDS,

368  8 FORMAT(a,4d25.16/(7x,5d25.16))
369  WRITE(6,8)' SUNVCT ',sunvct
370  WRITE(6,8)' VLIGHT ',vlight
371  WRITE(6,8)' SECPAU ',secpau
372  WRITE(6,8)' SRA ',sra
373  WRITE(6,8)' SD ',sd
374  WRITE(6,8)' CRA ',cra
375  WRITE(6,8)' CD ',cd
376  WRITE(6,8)' DPLXP ',dplxp
377 C
378  500 CONTINUE
379 C
380 C New computation of parallax partial w.r.t. inverse distance (parsecs)
381  Do j=1,3
382  e(j) = earth(j,1)
383  k(j) = star(j)
384  b0(j) = epbase(j,1)
385  v0(j) = epbase(j,2)
386  vpw2(j) = earth(j,2) + sitev(j,2)
387  Enddo
388 C Compute vector to Earth that is perpendicular to star vector
389  Call crosp(e, k, a)
390  Call crosp(k, a, b)
391 C Compute parallax correction to star vector at distance of 1 parsec
392 C (Radians)
393  dk(1) = b(1) / au_meters * convds
394  dk(2) = b(2) / au_meters * convds
395  dk(3) = b(3) / au_meters * convds
396 C print dK in arc-seconds
397 C WRITE(6,8)' dK: ', dK(1)/CONVDS, dK(2)/CONVDS,
398 C * dK(3)/CONVDS
399 C
400 C Convert to arc-sec - parsec units
401 C DPLXP(1) = DPLXP(1) * CONVDS
402 C DPLXP(2) = DPLXP(2) * CONVDS
403 C PUT the new parallax partials.
404 C CALL PUT4 ('PLX1PSEC ', DPLXP, 2, 1, 1 )
405 
406 C

408  t_prlx(1) = -(dotp(dk,b0)/vlight) / (1.d0 + (dotp(k,vpw2)/vlight))

411  * + ((dotp(k,v0)/vlight2) * dotp(dk,vpw2))
412  CALL put4 ('PLX1PSEC ', t_prlx, 2, 1, 1 )
