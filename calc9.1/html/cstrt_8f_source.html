3 C
4 C START initializes CALC. It reads the run title, the database experiment

7 C obtains the date and time of the CALC run from subroutine FDATE. It writes
8 C the CALC run title, the database experiment file to be processed, the run

479 C
480 C Subroutine to remove leading blanks in a character string and convert all
481 C letters to upper case.
482 C
483 C Input:
484 C Isize = Number of characters
485 C Iup = 0 ==> Don't convert to upper case
486 C Iup = 1 ==> Do convert to upper case
487 C Input/Output:
488 C String = The character string
489 C
490 C Programmer:
491 C 98.03.13 David Gordon Original program written
492 C
493  Integer*4 Isize, I, J, Icnt, Iup
494  Character*(*) String
495  Character*26 Upper_case, Lower_case
496  Data upper_case /'ABCDEFGHIJKLMNOPQRSTUVWXYZ'/
497  Data lower_case /'abcdefghijklmnopqrstuvwxyz'/
498 C
499  icnt = 0
500 C
501  write(6,'(i4,2x,a)') isize, string
502 C
503 C Get rid of leading blanks
504  20 Continue
505  icnt = icnt + 1
506  If (icnt .ge. isize) Return
507  If (string(1:1) .eq. ' ') Then
508  Do j=2,isize
509  string(j-1:j-1) = string(j:j)
510  Enddo
511  string(isize:isize) = ' '
512  Go to 20
513  Endif
514 C
515 C Special case, 'None', 'NONE', 'none', etc.
516  If (string(1:4) .eq. 'NONE') string(1:4) = 'None'
517  If (string(1:4) .eq. 'none') string(1:4) = 'None'
518 C
519  If (iup .ne. 1) Return
520 C Make all letters upper case
521  Do j=1,isize
522  Do i=1,26
523  If (string(j:j) .eq. lower_case(i:i)) Then
524  string(j:j) = upper_case(i:i)
525  Go to 30
526  Endif
527  Enddo
528  30 Continue
529  Enddo
530 C
531  Return
532  End

535  REAL*8 FUNCTION jdy2k (IYEAR, IMONTH, IDAY)
536  Implicit None
537 C
538 C Fucntion JDY2K: Function to convert year, month, day to full Julian
539 C day. The year can be either a four-digit year or a two-digit year.
540 C
541 C If a 4-digit year, this function is good from 1 March 1900 to 31
542 C December 2099.
543 C
544 C If a 2-digit year, this function is good from 1 January 1970 to
545 C 31 December 2069. If year is 70 - 99, 1900 is added. If year is
546 C 00 - 69, 2000 is added.
547 C
548 C Programmer:
549 C 98.07.23 D. Gordon Function written from code in cutcu.f
550 C
551  Integer*4 IYEAR, IMONTH, IDAY, IY, IM, ID
552 C
553  iy = iyear
554  im = imonth
555  id = iday
556 C
557  If (iy .ge. 70 .and. iy .le. 99) Then
558  iy = iy + 1900
559  Go To 100
560  Endif
561 C
562  If (iy .ge. 0 .and. iy .le. 69) Then
563  iy = iy + 2000
